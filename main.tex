\proofbanktitle{Exam Writeup Checklist}
\textbf{Algorithm:} input/output, data structs, return value.
\textbf{Correctness:} complete invariant/induction/exchange proof, not idea.
\textbf{DP:} State, Base, Transition, Order, Answer, Runtime, Space.
\textbf{Bounds:} define $n,V,E,W$; justify loops.
\textbf{Conventions:} $<$ vs $\le$, inclusive/exclusive, 0/1-index, reachable vs global negative cycle.

\proofbanktitle{Proof Bank: Generic Templates}
\proofitem{Loop inv}{Init true before loop. Maint: one step preserves. Term: invariant + stop condition $\Rightarrow$ answer.}

\proofitem{Induction/substitution}{Claim upper/lower bound with constants. State IH for all smaller inputs. Substitute IH. Collect terms. Pick constants so inequality holds. Base cases: choose $c$ large enough for all $n<n_0$.}

\proofitem{Substitution trap}{If $T(n)\le c\sum_i(a_in)^p+bn$ and $\sum_i a_i^p=1$, then $cn^p+bn$ FAILS. No slack absorbs $+bn$. Strengthen: $T(n)\le cn^p-dn^q$ with lower-order subtraction.}

\proofitem{Midterm recurrence}{$T(n)=T(4n/5)+T(3n/5)+an$. Guess $\Theta(n^2)$ because $(4/5)^2+(3/5)^2=1$. Lower: $T(n)\ge T(4n/5)+T(3n/5)\ge c((4n/5)^2+(3n/5)^2)=cn^2$. Upper must be strengthened: assume $T(m)\le cm^2-dm$ for all $m<n$. Then $T(n)\le c(4n/5)^2-d(4n/5)+c(3n/5)^2-d(3n/5)+an=cn^2-7dn/5+an\le cn^2-dn$ iff $d\ge5a/2$. Choose $d$ so, then choose $c$ large for base cases.}

\proofitem{D\&C}{Induct on size. Recursive calls solve smaller problems. Combine covers all optimum cases and constructs/chooses correct result.}

\proofitem{DP}{State meaning. Base correct. Induct in fill order. Transition enumerates all valid first/last choices; dependencies already optimal. Optimal solution uses one enumerated choice; every candidate feasible. Therefore $dp[s]$ optimum. Write answer $=dp[\dots]$.}

\proofitem{Parametric search}{Guess answer $M$. Build feasible$(M)$. Prove monotone: if $M$ works then any larger $M$ works (minimization). Binary search smallest feasible $M$. Correctness = binary search invariant + feasible predicate exact.}

\proofitem{Greedy}{Safe choice: for optimum $O$ containing current partial solution, if chosen $e\notin O$, exchange $f\in O$ with $e$ without worsening. Then induct.}

\proofbanktitle{Proof Bank: Sorting / Hash / DS}
\proofitem{BST inorder}{Induct on subtree. Left keys $<$ root, right keys $\ge$ root. IH sorts left/right. Left,root,right sorted and complete.}

\proofitem{Rank/interval}{Store $size=1+size(L)+size(R)$. Rank follows one search path; when going right add $size(L)+1$. Thus $O(h)$. $[a,b]=rank_{\le}(b)-rank_{<}(a)$.}

\proofitem{Heapify}{Children are heaps. Swap with larger child if violated; parent fixed. Only swapped child may violate; recurse to leaf. Cost height $O(\log n)$.}

\proofitem{BuildHeap}{Leaves are heaps. Bottom-up: children already heaps, so Heapify fixes subtree. Runtime: nodes height $h\le n/2^{h+1}$, total $\sum_h O(h)n/2^h=O(n)$.}

\proofitem{Sorting lower bound}{Comparison sort decision tree needs $\ge n!$ leaves. Worst-case comparisons $\ge$ height $\ge\log_2(n!)=\Omega(n\log n)$.}

\proofitem{Quicksort exp}{Pair compared at most once. $X_{ij}=1$ if ranks $i,j$ compared. Compare iff first pivot among ranks $i..j$ is $i$ or $j$, so $P(X_{ij})=2/(j-i+1)$. Sum $O(n\log n)$.}

\proofitem{Hashing}{Single key: for key $x$, $X_y=1[h(y)=h(x)]$. Universal hashing gives $E[X_y]\le1/m$, so expected chain collisions $\le(n-1)/m=\Theta(\alpha)$ and search $O(1+\alpha)$. Total collisions: $X=\sum_{i<j}X_{ij}$, $X_{ij}=1[h(x_i)=h(x_j)]$, hence $E[X]\le{n\choose2}/m=n(n-1)/(2m)=\Theta(n2/m)$. $k$-collision: $E[\#k\text{-collisions}]\le {n\choose k}/m^{k-1}$. Markov: $P[X\ge1]\le E[X]$.}
\proofitem{Hash mini}{Length-$\ell$ keys: pair collision prob $1/m^\ell$, so $E[\#collisions]={n\choose2}/m^\ell$; if $m^\ell\ge{n\choose2}$ then $P[\exists collision]\le1$. Random $q$-color vertices: edge mono prob $1/q$, so $E[\#mono]=|E|/q$.}
\proofitem{DSU}{Union by rank attaches smaller rank under larger. Rank increases only on equal-rank merge. Path compression preserves reps and shortens future paths. With both: $m$ ops in $O(m\alpha(n))$.}

\proofbanktitle{Proof Bank: Graph Traversal}
\proofitem{BFS}{Queue processed in nondecreasing distance. Discover $v$ from $u$: $d[v]=d[u]+1$. Shorter path would come from smaller-distance vertex processed earlier, so $v$ would already be discovered.}

\proofitem{DFS/topo}{Directed cycle iff DFS has back edge to GRAY ancestor. DAG has no back edges. For every edge $u\to v$, $f[u]>f[v]$; decreasing finish time is topo order.}

\proofitem{DFS edge facts}{Parenthesis theorem: finish intervals disjoint or nested; nested iff descendant. White-path theorem: $v$ descendant of $u$ iff when $u$ discovered, there is a white path $u\leadsto v$.}

\proofitem{SCC/Kosaraju}{Condensation graph is DAG. First DFS finish order puts source SCCs late. In reversed graph, DFS from highest finish cannot leave its SCC to an unvisited SCC. Each second-pass DFS outputs one SCC.}

\proofitem{SCC reachability pairs}{$u\leadsto v$ and $v\leadsto u$ iff $comp[u]=comp[v]$. For forbidden pair $(u,v)$ reject iff same component. Correct because SCC is maximal mutual reachability. Runtime $O(V+E)$.}

\proofitem{DAG SP/LP}{Topo order processes every predecessor before vertex. When processing $u$, $dist[u]$ already final over all paths ending at $u$; relaxing outgoing edges propagates optimum once. Runtime $\Theta(V+E)$.}

\proofbanktitle{Proof Bank: MST / Shortest Paths / Flow}
\proofitem{Cut property}{Let $e$ be light edge crossing cut respecting $A$. Take MST $T\supseteq A$ not using $e$. Add $e$ creates cycle with another crossing edge $f$. Since $w(e)\le w(f)$, $T-f+e$ is MST containing $A\cup\{e\}$.}

\proofitem{Cycle property}{If MST $T$ contains unique heaviest edge $e$ on cycle, remove $e$; cycle has another edge $f$ reconnecting cut with $w(f)<w(e)$. Then $T-e+f$ cheaper, contradiction.}

\proofitem{Kruskal}{DSU components define cut respecting chosen edges. Next accepted edge is lightest crossing some component cut. Cut property says safe. Induct until $|V|-1$ edges.}

\proofitem{Prim}{$S=$ vertices in tree. Heap chooses lightest edge crossing $(S,V\setminus S)$. Cut property says safe. Add one vertex; repeat.}

\proofitem{Dijkstra}{$S=$ finalized. Pick outside $x$ with min tentative dist. Any alternative path first crosses from $S$ to outside at $(u,v)$. Since $w\ge0$, path length $\ge d[u]+w(u,v)\ge d[v]\ge d[x]$. So $d[x]$ final.}

\proofitem{Bellman-Ford}{After $i$ passes, all shortest paths using $\le i$ edges are correct. Induct by last edge relaxation. If no negative cycle, shortest simple path has $\le V-1$ edges. If pass $V$ improves, reachable negative cycle.}

\proofitem{Max-flow}{If no augmenting path exists, let S be vertices reachable from s
in residual graph. For every edge $u\in S$,$v\notin S$: $cf(u,v)=0$, therefore $f(u,v)=c(u,v)$.
For every edge $u\notin S$,$v\in S$:
$cf(v,u)=0$, therefore $f(u,v)=0$.

Hence net flow crossing cut equals cut capacity:
$|f| = c(S,T)$.

Since every flow $\le$ every cut,
flow and cut are both optimal.}

\proofitem{Flow cert}{Any flow value $\le$ any cut capacity. Flow value $F$ + cut capacity $F$ proves both optimal.}

\proofitem{Bipartite matching flow}{Caps 1 force each left/right vertex used at most once. Integral capacities give integral max flow. Unit flow through $l\to r$ is matched pair; every matching gives same-size flow.}

\proofitem{Cap assignment flow}{Build $s\to$ people cap 1, person$\to$ acceptable item cap 1, item$\to t$ cap capacity. Feasible iff maxflow = \#people. Integral flow = assignment; assignment = integral flow.}

\proofitem{SP transform}{Multiplying every edge weight by $a>0$ preserves all shortest paths. Adding $a$ to every edge does not: paths with different number of edges get different extra cost.}

\proofitem{Resource SP}{State graph $(v,b)$ where $b=$ used budget / remaining battery. Add transition only if resource feasible. Run BFS/Dijkstra depending on edge costs. Size $O(VB+EB)$. Correct since every feasible walk corresponds to one state path and vice versa.}
\proofitem{2D sorted matrix}{Rows/cols sorted. Start top-right. If $A[i,j]>x$, move left; if $A[i,j]<x$, move down. Each step deletes one impossible row/col, so $O(n+m)$, $O(1)$.}
\proofbanktitle{Proof Bank: Trees / Exam Reductions}
\proofitem{Tree DP}{Postorder. Return height/best info from children. Best at $u$ is max(best in left, best in right, best crossing/using $u$). Induct on subtree: any optimal object either lies fully in one child or uses $u$.}

\proofitem{Tree diameter}{Let $h(u)=1+\max(h_L,h_R)$, empty height $-1$. $diam(u)=\max(diam_L,diam_R,h_L+h_R+2)$. Path either inside one child or crosses through $u$.}
\proofitem{Root path}{Longest simple path forced to pass through root: $h(root.left)+h(root.right)+2$ with empty height $-1$. Compute heights postorder in $O(n)$.}
\proofitem{Tree rebuild}{Inorder+postorder: root=last(post); split inorder at root; subtree sizes split postorder; recurse. Preorder+inorder: root=first(pre). Pre+post alone not unique.}

\proofitem{Unicyclic spanning trees}{Connected graph with $|E|=|V|$ has exactly one cycle $C$. Every spanning tree = delete exactly one edge of $C$. Deleting non-cycle edge disconnects; deleting cycle edge breaks only cycle. Hence \#spanning trees $=|C|$.}


\proofbanktitle{Proof Bank: Online / Random}

\proofitem{LRU paging}{Cache size $k$. Phase has $\le k$ distinct pages; new phase begins at request of $(k+1)$-st distinct. LRU: $\le k$ misses/phase, because requested page survives until $k$ other distinct pages requested. OPT: $\ge1$ miss/completed phase, because phase + first request of next phase contains $k+1$ distinct pages but cache holds $k$. Thus LRU $\le k\cdot$OPT$+O(k)$; LRU is $k$-competitive.}
\sectiontitle{Complexity / Asymptotics}

$\log n \ll \sqrt n \ll n \ll n\log n \ll n^2 \ll n^3 \ll 2^n \ll n! \ll n^n$

$O(g):f\le cg$ eventually. $\Omega(g):f\ge cg$. $\Theta(g)=O\cap\Omega$.
$o(g):f/g\to0$. $\omega(g):f/g\to\infty$.
Stirling: $n!\sim\sqrt{2\pi n}(n/e)^n$, so $\log(n!)=\Theta(n\log n)$.

\textbf{Useful sums:}

$\sum_{i=1}^n i=\frac{n(n+1)}2=\Theta(n^2),\;
\sum_{i=0}^{\log n}n=n\log n,\;
\sum_{i\ge0}2^{-i}=2,\;
\sum_{h\ge0}h2^{-h}=2$

\textbf{Master / recurrence solver:}

\textbf{Master:} $T(n)=aT(n/b)+f(n)$. Let $d=\log_b(a)$.
Compare $f(n)$ with $n^d$.

\begin{tabular}{@{}ll@{}}
$f$ smaller: $f=O(n^{d-\varepsilon})$ & $T=\Theta(n^d)$\\
same: $f=\Theta(n^d \log^k n)$ & $T=\Theta(n^d \log^{k+1}n)$\\
$f$ bigger: $f=\Omega(n^{d+\varepsilon})$ & $T=\Theta(f)$ if regularity\\
\end{tabular}

\textbf{Fast mental version if $f=n^k$:}
\begin{tabular}{@{}lll@{}}
$k<d$ & leaves win & $\Theta(n^d)$\\
$k=d$ & tie/layers equal & $\Theta(n^d \log n)$\\
$k>d$ & root/work wins & $\Theta(n^k)$\\
\end{tabular}

\textbf{Non-equal splits:} $T=\sum_i a_i T(b_i n)+\Theta(n^k)$.
Find $p$ s.t. $\sum_i a_i b_i^p = 1$.

\subsectiontitle{Hash Tables}

\textbf{Idea:} bucket $h(key)\bmod m$. \textbf{Collisions:} chaining / probing. \textbf{Load factor} $\alpha=n/m$.

Hash ops: insert/delete expected O(1), search expected O(1+alpha), all worst O(n); m=Theta(n) => alpha=Theta(1).
\textbf{Expected total collisions:}
$X=\sum_{i<j}X_{ij}$, where $X_{ij}=1[h(x_i)=h(x_j)]$.
Universal hashing gives P[Xij=1]<=1/m, so E[X]=sum E[Xij] <= C(n,2)/m = n(n-1)/(2m)=Theta(n2/m).
For one key: $\mathbb E[\#\text{collisions}]\le(n-1)/m=\Theta(\alpha)$, hence expected search $O(1+\alpha)$.
\textbf{Traps:} HashMap/HashSet $O(1)$ is expected/amortized, not worst-case.
Bad hashing / all collisions $\Rightarrow O(n)$.

\textbf{k-collision stencil:} $X_S=1$ if fixed $k$-set collides. $P[X_S=1]\le1/m^{k-1}$. $E[X]=\sum_{|S|=k}E[X_S]\le {n\choose k}/m^{k-1}$. If need existence prob: $P[X\ge1]\le E[X]$.

\subsectiontitle{Rod Cutting}

Rod cutting: revenue[0]=0; revenue[l]=max over 1<=c<=l of price[c]+revenue[l-c].

fn rod_cut(price:Vec<i32>, n:usize){
  revenue=vec![0;n+1]; first=vec![0;n+1];
  for length in 1..=n {revenue[length]=-INF;
    for cut in 1..=length {
      candidate=price[cut]+revenue[length-cut];
      if candidate>revenue[length] {
        revenue[length]=candidate; first[length]=cut
      }
    }

\subsectiontitle{Matrix Chain Multiplication}
Matrix $i$: $p_i\times p_{i+1}$. $dp[i][j]=$ min cost for $A_i\cdots A_j$.
MCM recurrence: dp[i][i]=0; dp[i][j]=min over i<=k<j of dp[i][k]+dp[k+1][j]+p[i]*p[k+1]*p[j+1].
\begin{lstlisting}[aboveskip=0pt,belowskip=0pt]
fn matrix_chain(p:Vec<usize>){     // optimal parenthesization
  n=p.len()-1; dp=vec![vec![0;n];n]; split=vec![vec![0;n];n];
  for len in 2..=n {for i in 0..=n-len {
    j=i+len-1; dp[i][j]=INF;
    for k in i..j {

    }
  }}

}

\subsectiontitle{Optimal BST}
Sorted keys $K_i$ with successful probs $p_i$; optional unsuccessful $q_i$. $e[i][j]=$ min expected cost for subtree $i..j$, $w[i][j]=$ prob mass, $root[i][j]=$ best root.
OBST: e[i][i-1]=0, w[i][i-1]=q[i-1]; e[i][j]=min over root r in [i,j] of e[i][r-1]+e[r+1][j]+w[i][j].
\begin{lstlisting}[aboveskip=0pt,belowskip=0pt]
fn optimal_bst(p,q,n){
  for i in 1..=n+1 {e[i][i-1]=0; w[i][i-1]=q[i-1]}
  for len in 1..=n {for i in 1..=n-len+1 {
    j=i+len-1; e[i][j]=INF; w[i][j]=w[i][j-1]+p[j]+q[j];
    for r in i..=j {
      cost=e[i][r-1]+e[r+1][j]+w[i][j];
      if cost<e[i][j] {e[i][j]=cost; root[i][j]=r}
    }
  }}
  (e,root)
}
\end{lstlisting}
Runtime Theta(n\textasciicircum3), space Theta(n\textasciicircum 2).

\subsectiontitle{Knapsack 0/1}
$dp[i][c]=$ best value using first $i$ items, capacity $c$.
0/1 knapsack: dp[i][c]=max(skip dp[i-1][c], take v[i-1]+dp[i-1][c-w[i-1]]) if w[i-1]<=c.
\begin{lstlisting}[aboveskip=0pt,belowskip=0pt]
fn knapsack(w:Vec<usize>, v:Vec<i32>, cap:usize){ // choose item at most once
  n=w.len(); dp=vec![vec![0;cap+1];n+1];
  for i in 1..=n {for c in 0..=cap {
    dp[i][c]=dp[i-1][c];
    if w[i-1]<=c {
      dp[i][c]=dp[i][c].max(v[i-1]+dp[i-1][c-w[i-1]])
    }
  }}
  dp[n][cap]
}
// 1D 0/1: capacity descending. Unbounded: capacity ascending.
\end{lstlisting}

\subsectiontitle{Cut / Cycle Properties}
Cut $(S,V-S)$: edge crosses cut iff one endpoint in $S$, one outside. Light edge = minimum-weight crossing edge.

\textbf{Cut property:} if cut respects chosen edge set $A$, then a light crossing edge is safe.

\textbf{Cycle property:} for any cycle, a heaviest edge on that cycle is excluded from some MST. If weights distinct, the heaviest edge is in no MST.

\sectiontitle{Flow Networks}

\subsectiontitle{Max-Flow Min-Cut / Reductions}
Max-flow min-cut: $\max |f|=\min c(S,V\setminus S)$, $s\in S,t\notin S$.
Optimality certificate: flow value $F$ and cut capacity $F$.

\sectiontitle{Randomized / Probabilistic Analysis}
Indicator: $X_A=\mathbf 1[A]$, $\mathbb E[X_A]=\Pr(A)$. Linearity: $E[\sum_i X_i]=\sum_i E[X_i]$.
No independence needed.

Hiring: $P(i\text{ new best})=1/i$, so $E[\#hires]=\sum_i1/i=H_n=\Theta(\log n)$. Hat-check: $P(i\text{ own hat})=1/n$, so $E[\#fixed]=1$.

Randomized quicksort: random pivot avoids input-dependent bad order; expected $\Theta(n\log n)$, worst still $\Theta(n^2)$.

\textbf{Birthday:} $\Pr[\text{all different among }q\text{ into }m]=\prod_{i=0}^{q-1}(1-i/m)\approx e^{-q(q-1)/(2m)}$; for 365 days, $q=23$ gives match prob $>1/2$.
Integral bound for decreasing positive $f$: $\sum_{k=a+1}^{b}f(k)\le\int_a^b f(x)\,dx$.
\sectiontitle{Online Algorithms / Competitive Analysis / Experts}
Online: decide before future known.
Competitive ratio: $ALG(I)\le r\,OPT(I)+c$.
\textbf{Ski rental:} rent 1/day, buy B. Rent until paid B, then buy. If $T\le B$: ALG=T=OPT. If $T>B$: ALG<=B+B=2B=2OPT. Lower bound: adversary stops just before/after purchase; deterministic CR=2.
\subsectiontitle{Caching MC traps}
LRU evicts least-recently requested page. LFU evicts smallest request count; tie usually LRU. Neither has pointwise fewer misses than the other on every sequence. LRU is $k$-competitive for cache size $k$; no deterministic paging algorithm beats $k$ in worst-case competitive ratio.

\subsectiontitle{Experts / Weighted Majority}
$n$ experts, rounds $t=1..T$. Expert $i$ predicts; loss $\ell_{i,t}\in\{0,1\}$.
Weights $w_i=1$ initially. Let $W=\sum_iw_i$.

\begin{lstlisting}[aboveskip=0pt,belowskip=0pt]
for t in 1..=T {
  predict weighted majority of experts;
  observe true outcome;
  for expert i wrong this round {
    w[i] *= beta;        // 0 < beta < 1, often beta=1/2
  }
}
\end{lstlisting}

Weighted prediction: choose label with larger total weight.
Potential: $W_t=\sum_i w_i^{(t)}$.
If ALG makes mistake, wrong-weight $\ge W_t/2$, so
$W_{t+1}\le W_t/2\cdot\beta+W_t/2=W_t(1+\beta)/2$.
If best expert has $m^*$ mistakes, its weight is $\beta^{m^*}$, hence
$\beta^{m^*}\le W_T\le n\big((1+\beta)/2\big)^M$.
For $\beta=1/2$: $M\le 2.41(m^*+\log_2 n)$.
General randomized Hedge: choose expert with prob $w_i/W$; update $w_i\leftarrow w_i e^{-\eta \ell_{i,t}}$; regret $O(\sqrt{T\log n})$.
\subsectiontitle{Prophet Inequality}
Independent nonnegative $X_i$, known distributions. $M=\max_iX_i$, $z=E[M]$, threshold $\tau=z/2$. Accept first $X_i\ge\tau$. Guarantee $E[ALG]\ge z/2$. Proof stencil: $q=P[\forall i:X_i<\tau]$, $b_i=E[(X_i-\tau)^+]$. Value from threshold part $\ge\tau(1-q)$; overshoot part $\ge q\sum_i b_i$; combine to get $\ge z/2$.

Secretary: random order; observe first $n/e$ candidates, remember best rank seen, then hire first later candidate better than all observed.
\[
P(\text{success})\approx 1/e
\]

\sectiontitle{WORKED EXAM PROBLEMS}

\workedq{P01}{Loop invariant proof: linear search returns matching index or NIL.}{
\inv before iteration $i$, no earlier position contains $v$: $A[j]\ne v$ for all $j<i$. \init $i=1$: empty set. \maint if $A[i]\ne v$, then the statement extends to all $j<i+1$. \term if equality found, returned index correct; if loop finishes, invariant says no position contains $v$.
}

\workedq{P02}{Asymptotic proof with constants: show $(n+a)^b=\Theta(n^b)$ for fixed constants.}{
For $n\ge2|a|$: $n/2\le n+a\le 2n$. Raise to fixed $b>0$: $2^{-b}n^b\le(n+a)^b\le2^bn^b$. Choose constants $c_1=2^{-b},c_2=2^b,n_0=2|a|$.
}

\workedq{P03}{Recursion tree: $T(n)=8T(n/2)+\Theta(n^2)$, leaves dominate.}{
Level $i$: $8^i$ nodes, size $n/2^i$, work $8^i c(n/2^i)^2=cn^2 2^i$. Sum until $\lg n$: $cn^2\sum_{i=0}^{\lg n}2^i=\Theta(n^3)$. Leaves: $8^{\lg n}=n^3$. Hence $\Theta(n^3)$.
}

\workedq{P04}{Recursive insertion sort: remove last element, sort prefix, insert last.}{
Recurrence $T(n)=T(n-1)+\Theta(n)$. Unroll: $T(n)=\Theta(\sum_{i=1}^n i)=\Theta(n^2)$. Correctness by induction: sorted prefix by IH; insertion puts last element in unique sorted position.
}

\workedq{P05}{Strengthened recurrence proof: split sizes $3n/5,4n/5$ plus linear work.}{
$T(n)=T(3n/5)+T(4n/5)+an$. Guess $\Theta(n^2)$ since $(3/5)^2+(4/5)^2=1$. Upper needs slack: IH $T(m)\le cm^2-dm$. Then $T(n)\le cn^2-d(7n/5)+an\le cn^2-dn$ iff $d\ge5a/2$. Pick $d$, then pick $c$ for bases. Lower: $T(n)\ge c((3n/5)^2+(4n/5)^2)=cn^2$.
}

\workedq{P06}{Master/Akra pattern: choose fastest among several recurrence-defined algorithms.}{
Compute exponent. $aT(n/b)+n^k$: compare $k$ to $\log_ba$. Examples: $2T(n/2)+n^3\to\Theta(n^3)$; $30T(n/4)+n\to\Theta(n^{\log_4 30})$; $8T(n/2)+1\to\Theta(n^3)$; $T(n/2)+n^4\to\Theta(n^4)$; $2T(n-1)+1\to\Theta(2^n)$. Pick smallest asymptotic.
}

\workedq{P07}{Restaurant spacing DP: choose subset of line positions at mutual distance at least $k$.}{
Sort by position. $dp[i]=$ max profit using restaurants $i..n$. $next(i)=$ first $j>i$ with $M_j\ge M_i+k$. $dp[n+1]=0$, $dp[i]=\max(dp[i+1],p_i+dp[next(i)])$. \ans $dp[1]$. Proof: optimum skips $i$ or takes $i$, then all remaining choices start at $next(i)$. Runtime $O(n\log n)$ with binary search.
}

\workedq{P08}{Restaurant variant: force the current restaurant to be the leftmost open one.}{
$c[i]=$ best profit among solutions whose leftmost chosen restaurant is $i$. Then $c[i]=p_i+\max\{c[j]:j>i, M_j\ge M_i+k\}$, with empty max $0$. \ans $\max_i c[i]$. Use suffix maxima plus binary search for $O(n\log n)$, or scan for $O(n^2)$.
}

\workedq{P09}{Binary search proof: sorted array membership in $O(\log n)$.}{
Maintain half-open interval $[l,r)$ with invariant: if $x$ is present then it lies in $A[l..r)$. Compare with $A[m]$. If $A[m]<x$, discard $[l,m]$; if $A[m]>x$, discard $[m,r)$. Sortedness preserves invariant. Termination gives found index or empty candidate set. Runtime $T(n)=T(n/2)+O(1)$.
}

\workedq{P10}{2D sorted matrix: rows and columns increasing, find $x$ in $O(n)$.}{
Start at top-right $(i=1,j=n)$. If $A[i,j]=x$ return. If $A[i,j]>x$, move left: everything below in column $j$ is also $>x$. If $A[i,j]<x$, move down: everything left in row $i$ is also $<x$. Each step removes one row/column, so at most $2n-1$ steps.
}

\workedq{P11}{Augmented BST interval counting: count keys in $[a,b]$ with subtree sizes.}{
Store $size(x)=1+size(L)+size(R)$ and update on insert/delete rotations/path. $rank_{<}(x)$ follows search path; when moving right add $size(left)+1$. $rank_{\le}(x)$ similarly includes equals. Count $[a,b]=rank_{\le}(b)-rank_{<}(a)$. Runtime $O(h)$.
}

\workedq{P12}{MinStack: stack with Push, Pop, GetMin all worst-case $O(1)$.}{
Use two stacks: data stack $S$, minimum stack $M$. Push $x$: push to $S$; if $M$ empty or $x\le top(M)$ push $x$ to $M$. Pop: $y=pop(S)$; if $y=top(M)$ pop $M$. GetMin returns $top(M)$. Duplicates need $\le$ on push, not just $<$.
}

\workedq{P13}{Heap / priority queue trace: Build-Max-Heap and Heapify correctness.}{
BuildHeap calls Heapify from $\lfloor n/2\rfloor$ down to 1. Invariant: before processing $i$, all nodes $>i$ root heaps. Heapify assumes children heaps and pushes violation downward. Runtime $\sum_h (n/2^{h+1})O(h)=O(n)$, not $O(n\log n)$.
}

\workedq{P14}{Comparison sorting lower bound: prove $\Omega(n\log n)$ comparisons.}{
Any deterministic comparison sort is a decision tree. Each permutation of $n$ distinct keys must end in a different leaf, so at least $n!$ leaves. Height $h$ satisfies $2^h\ge n!$, so $h\ge\log_2(n!)=\Omega(n\log n)$.
}

\workedq{P15}{Randomized Quicksort comparison probability relative to a fixed element.}{
For ranks $i<j$, they are compared iff the first pivot chosen among ranks $i..j$ is either $i$ or $j$. Thus $P(i,j\text{ compared})=2/(j-i+1)$. For element of rank $r$, probability with rank $q$ is $2/(|q-r|+1)$; sort options by distance from $r$.
}

\workedq{P16}{All-equal Lomuto partition: bad split and robust duplicate fix.}{
If partition condition is $\le pivot$, all equal elements move left, pivot index returns $r$. Recurrence $T(n)=T(n-1)+\Theta(n)=\Theta(n^2)$. Fix with 3-way partition into $<,=,>$ pivot or choose middle of equal block.
}

\workedq{P17}{Simple uniform hashing: expected number of pair collisions.}{
$X=\sum_{i<j}X_{ij}$, $X_{ij}=1[h(i)=h(j)]$. $E[X_{ij}]=1/m$. Thus $E[X]={n\choose2}/m=\Theta(n^2/m)$. For collisions with first slot only: pair probability $1/m^2$, so expectation ${n\choose2}/m^2=\Theta(n^2/m^2)$.
}

\workedq{P18}{Hash values are length-$\ell$ arrays over $m$ symbols: expected collisions at most 1.}{
For fixed distinct $x,y$, collision requires equality in all $\ell$ coordinates, so $P[x,y\text{ collide}]=(1/m)^\ell$. Let $X=\sum_{x<y}X_{xy}$. Then $E[X]={n\choose2}/m^\ell\le n^2/(2m^\ell)\le1$ if $m^\ell\ge n^2$.
}

\workedq{P19}{Hash chains high-probability bound: no bucket longer than $\ell$.}{
A $k$-collision is a $k$-set in one bucket. $E[\#k\text{-collisions}]\le {n\choose k}/m^{k-1}\le n^k/m^{k-1}$. Chain length $>\ell$ iff an $(\ell+1)$-collision exists. If $m\ge100n^{1+1/\ell}$, then $E[\#(\ell+1)\text{-collisions}]\le n^{\ell+1}/m^\ell\le1/100^\ell\le0.01$. Markov: bad prob $\le0.01$.
}

\workedq{P20}{Expected hires with random arrival, skip first 10 percent.}{
For position $i>n/10$, indicator $X_i=1$ if candidate $i$ is best among first $i$. $P[X_i=1]=1/i$. Expected hires $=\sum_{i=n/10+1}^n1/i\le\int_{n/10}^n dx/x=\ln 10<10$.
}

\workedq{P21}{Random edge/color expectation: count bad local events by indicators.}{
If each edge/vertex color is independent uniform among $q$ colors, define $X_e=1$ if event $e$ is bad. Usually $P[X_e=1]$ is $1/q$ or $1/q^r$. Then $E[\#bad]=\sum_eP[X_e=1]$. Use Markov $P[\#bad\ge1]\le E[\#bad]$ for existence/probability bounds.
}

\workedq{P22}{Bipartite matching as unit-capacity max-flow.}{
Build $s\to L$ cap 1; original edges $L\to R$ cap 1; $R\to t$ cap 1. Integral capacities give integral max flow. Capacity 1 forces each vertex used once. Flow edge $l\to r$ is a matched pair; every matching gives same-size flow. Hence max flow = max matching.
}

\workedq{P23}{Capacitated activity/cheese assignment: each person gets one acceptable item, bins have capacities.}{
Network: $s\to$ each person/type cap demand; person/type$\to$item if acceptable cap INF or demand; item$\to t$ cap supply/spots. Feasible iff maxflow equals total people/students. Integral flow gives integer assignment; assignment gives flow.
}

\workedq{P24}{Domino tiling / maximum non-overlapping adjacent grid pairs.}{
Checkerboard color cells. Black cells $L$, white cells $R$. For every adjacent allowed pair add edge $L\to R$. Max domino placement = maximum bipartite matching: each domino is one matched edge and matching prevents cell reuse.
}

\workedq{P25}{Edge-disjoint paths: route multiple paths without reusing edges.}{
Give every edge capacity 1 and run maxflow. Value $k$ iff there are $k$ edge-disjoint paths (integrality). For undirected edges, either replace by two directed arcs sharing one unit-capacity gadget, or orient both ways carefully with a single capacity-1 edge gadget.
}

\workedq{P26}{Vertex-disjoint paths / simple path through required vertex $z$.}{
To find simple path $x\leadsto z\leadsto y$, need internally vertex-disjoint paths from $z$ to $x$ and $z$ to $y$. Split every vertex $v$ into $v_{in}\to v_{out}$ cap 1 (INF for terminals if allowed). Original edge $u-v$ becomes arcs $u_{out}\to v_{in}$ and $v_{out}\to u_{in}$. Add $x,y\to t$ cap 1, source $z$, ask maxflow value 2.
}

\workedq{P27}{Special required edge $e^*=\{a,b\}$ in an edge-simple $u$-$v$ trail.}{
Try both orientations. For orientation $a\to b$, remove $e^*$ and require edge-disjoint prefix/suffix paths $u\leadsto a$ and $b\leadsto v$ using unit edge capacities; similarly orientation $b\to a$. If flow construction returns both paths, concatenate prefix + $e^*$ + suffix. If both orientations fail, no valid trail.
}

\workedq{P28}{Haunted-house escape: $k$ agents, grid over times, ghosts, no two agents same cell.}{
Build time-expanded flow. For every safe cell $(c,t)$ create $c^t_{in}\to c^t_{out}$ cap 1. For each legal stay/move $c\to c'$ add $c^t_{out}\to c'^{t+1}_{in}$ cap 1 if target safe. Source to start cells at time 1 cap 1. Boundary cells at times $\le T$ connect to sink cap 1. Maxflow $=k$ iff all escape. Size $O(Tn^2)$ nodes/edges up to constants.
}

\workedq{P29}{Shortest path with battery/resource budget: minimize time, spend at most $B$.}{
State graph has vertices $(v,b)$ = at $v$ after spending $b$ battery, $0\le b\le B$. For edge $e=(u,v)$ add $(u,b)\to(v,b+b(e))$ with weight $t(e)$ if $b+b(e)\le B$ (both directions if undirected). Run Dijkstra from $(s,0)$. \ans $\min_{b\le B}dist(t,b)$. Runtime $O(BE\log(BV))$.
}

\workedq{P30}{Road network mutual reachability forbidden for listed directed pairs.}{
Compute SCCs with Kosaraju/Tarjan in $O(V+E)$. For each forbidden pair $(u,v)$, reject iff $comp[u]=comp[v]$. Same SCC iff $u\leadsto v$ and $v\leadsto u$. If all pairs pass, accept. Given $\ell\le E$, total runtime $O(V+E)$.
}

\workedq{P31}{Topological-sort / Course Schedule: can all prerequisites be satisfied?}{
Directed graph prerequisite $u\to v$. Run DFS colors or Kahn indegrees. If cycle/back edge exists, impossible. Otherwise topo order exists and gives valid schedule. Runtime $O(V+E)$. Proof: DAG iff topo order iff no DFS back edge.
}

\workedq{P32}{DAG maximum-weight path with vertex weights from $s$ to $t$.}{
Topologically sort. Initialize $dp[s]=w(s)$, others $-\infty$. In topo order, relax $u\to v$: $dp[v]=\max(dp[v],dp[u]+w(v))$. \ans $dp[t]$. Correct because all predecessors of $v$ processed before $v$. Runtime $O(V+E)$.
}

\workedq{P33}{Verify claimed shortest-path distances in graph with negative edges but no negative cycles.}{
Check $d[s]=0$. For every edge $(u,v)$ check triangle inequality $d[v]\le d[u]+w(u,v)$; if violated reject. Build tight-edge graph $E'=\{(u,v):d[v]=d[u]+w(u,v)\}$. DFS/BFS from $s$ in $E'$. Accept iff all vertices reached. Inequality gives $d[v]\le$ every path length; tight paths give actual paths of length $d[v]$, so equality.
}

\workedq{P34}{Bellman-Ford negative-cycle / altitude consistency with signed edge changes.}{
A bad closed tour is a directed cycle with nonzero total weight. Run Bellman-Ford on $w$ and on $-w$ from a super-source connected to all vertices by 0 edges. A negative cycle in $w$ or in $-w$ means some cycle has nonzero sum. If neither run detects, all cycles have sum 0. Runtime $O(VE)$.
}

\workedq{P35}{Shortest-path weight transform MCQ: add constant vs multiply by positive constant.}{
Multiplying all edge weights by $a>0$ preserves every path ordering: $w(P)$ becomes $a w(P)$. Adding $a$ to every edge changes $w(P)$ to $w(P)+a\cdot |P|$, so paths with different numbers of edges can change order.
}

\workedq{P36}{Dijkstra choice proof: nonnegative weights only.}{
Maintain finalized set $S$. Pick outside vertex $x$ with minimum tentative distance. Any $s\leadsto x$ path first crosses from $S$ to $V\setminus S$ at $(u,v)$; length to $v$ is at least $d[u]+w(u,v)\ge d[v]\ge d[x]$ because $w\ge0$. Thus $d[x]$ is final. Negative edges break this.
}

\workedq{P37}{Bellman-Ford exact invariant and negative-cycle certificate.}{
After $i$ full passes, $d[v]$ is shortest path length among paths using at most $i$ edges. Induct by last edge. Without negative cycles, shortest paths can be simple, so use at most $V-1$ edges. If a $V$-th pass still relaxes, there is a reachable negative cycle.
}

\workedq{P38}{MST over complete graph of points: connect all islands with $n-1$ routes.}{
Build complete undirected graph on the $n$ points with edge weight squared Euclidean distance. Need connected graph with $n-1$ routes minimizing total cost = MST. Run Kruskal/Prim on $O(n^2)$ edges. Runtime $O(n^2\log n)$; output MST edges.
}

\workedq{P39}{MST cut/cycle proof: min crossing edge safe, max cycle edge unsafe.}{
Cut: add lightest crossing edge $e$ to an MST not using it; cycle contains another crossing edge $f$ with $w(f)\ge w(e)$; swap $f\mapsto e$. Cycle: if unique heaviest edge $e$ of a cycle is in MST, remove it; another cycle edge reconnects the cut cheaper, contradiction.
}

\workedq{P40}{Minimum-weight feedback edge set in undirected weighted graph.}{
Need remove edges so remaining graph is acyclic. Equivalent: keep an acyclic subset of maximum total weight. In connected graph, max kept acyclic set is a maximum spanning tree. Run Kruskal in descending weights, keep tree edges, remove all others. Correct because any acyclic connected kept set has at most $V-1$ edges and maximum spanning tree maximizes kept weight.
}

\workedq{P41}{Airport + roads connectivity: cities can connect by roads or both having airports.}{
Add super-node $r$. For every city $v$, add edge $(r,v)$ of cost airport cost $c_v$. Keep original road edges with costs $w_e$. Run MST on augmented graph. Chosen original edges = roads; chosen $r$-edges = airports. Super-node models free travel between airports. Runtime $O((E+V)\log(E+V))$.
}

\workedq{P42}{Edge belongs to at least one MST: linear-time query for edge $e=(a,b)$.}{
Let $W=w(e)$. Run DFS/BFS from $a$ using only edges with weight $<W$. If $b$ reached, then $e$ is heaviest on a strictly lighter cycle, so no MST contains it. If not reached, take component $S$ of $a$ in subgraph $w<W$; every edge crossing $(S,V\setminus S)$ has weight $\ge W$, and $e$ crosses, so $e$ is lightest for this cut. Thus some MST contains $e$. Runtime $O(V+E)$.
}

\workedq{P43}{Unique MST vs many MSTs: detect swappable equal-weight non-tree edge.}{
Compute one MST $T$. For every non-tree edge $e=(u,v)$, find maximum edge weight $M$ on the unique path $u\leadsto v$ in $T$. If $w(e)=M$, swap $e$ with a max edge on that path to get another MST: MANY. If all $w(e)>M$, no equal swap exists: ONE. Naive runtime $O(EV)$; faster with LCA if needed.
}

\workedq{P44}{Connected graph with $n$ vertices and $n$ edges: count spanning trees in $O(n)$.}{
Connected with $|E|=|V|$ has exactly one cycle. Find cycle length $c$ by DFS back edge or repeatedly delete degree-1 leaves until the cycle remains. Every spanning tree is obtained by deleting exactly one cycle edge; deleting non-cycle edge disconnects. \ans number of spanning trees $=c$, hence between 3 and $n$ for simple graph.
}

\workedq{P45}{Tree diameter: output farthest pair of nodes in binary tree.}{
Postorder returns $(height,deepest,bestLen,bestPair)$. At node $u$, candidate crossing pair is deepest node in left subtree to deepest node in right subtree with length $h_L+h_R+2$ (handle empty child). Best is max of left best, right best, crossing. Runtime $O(n)$.
}

\workedq{P46}{Longest simple path constrained to pass through the root.}{
Compute heights of left and right subtrees. If height counts edges and empty height is $-1$, answer is $h(root.left)+h(root.right)+2$. The path through root is maximized by independently choosing deepest descendant on each side. Runtime $O(n)$.
}

\workedq{P47}{Reconstruct binary tree from inorder and postorder traversals.}{
Root is last element of postorder. Find root in inorder; left/right inorder blocks determine subtree sizes. Split postorder into corresponding left/right blocks, recurse. Base empty list. Runtime $O(n)$ with value$\to$inorder-index map. Preorder+inorder similar; preorder+postorder alone is not unique.
}

\workedq{P48}{Travel passes DP: cover sorted travel days with 1/7/30-day passes.}{
Days $d_1<\cdots<d_n$, lengths $L$, costs $c_L$. $dp[i]=$ min cost to cover days $i..n$. $next(i,L)=$ first $j$ with $d_j\ge d_i+L$. $dp[n+1]=0$, $dp[i]=\min_L(c_L+dp[next(i,L)])$. Fill decreasing $i$. \ans $dp[1]$. Trap: pass bought on day $d_i$ covers days $<d_i+L$.
}

\workedq{P49}{Weighted interval scheduling: choose non-overlapping intervals of max weight.}{
Sort by finish. $p(i)=$ last interval ending $\le start_i$. $dp[i]=$ best using first $i$ intervals. $dp[0]=0$, $dp[i]=\max(dp[i-1],w_i+dp[p(i)])$. \ans $dp[n]$. Runtime $O(n\log n)$ with binary search for $p$.
}

\workedq{P50}{Cardinality-constrained weighted intervals: choose at most $k$ disjoint intervals.}{
Sort by finish and compute $p(i)$. $dp[i][j]=$ best value using first $i$ intervals and at most $j$ intervals. Base $dp[0][j]=dp[i][0]=0$. Recurrence $dp[i][j]=\max(dp[i-1][j], w_i+dp[p(i)][j-1])$. \ans $dp[n][k]$. Runtime $O(nk+n\log n)$.
}

\workedq{P51}{Interval scheduling greedy: max number of mutually disjoint intervals.}{
Sort by increasing finish time. Repeatedly take first interval compatible with current last finish. Exchange proof: earliest-finishing interval $e$ can replace first interval of any optimum without reducing feasibility. Then recurse on intervals starting after $e$.
}

\workedq{P52}{Divide farm into exactly 3 contiguous nonempty parts minimizing largest sum.}{
Positive $A_i\le1000$. Binary search answer $M$ between $\max A_i$ and $\sum A_i$. Feasible$(M)$: greedily scan and start new part when adding next block would exceed $M$; let parts count be $q$. Feasible iff $q\le3$ (then split further to get exactly 3 nonempty parts). Monotone predicate. Since sum $\le1000n$, runtime $O(n\log n)$. Recover cuts by greedy under optimal $M$ and split if fewer than 3 parts.
}

\workedq{P53}{Split array into $k$ contiguous parts minimizing maximum load.}{
Same as farm with $k$ instead of 3. Lower bound $L=\max a_i$, upper $R=\sum a_i$. Greedy feasibility uses minimum possible number of parts for capacity $M$. Binary search smallest feasible $M$. Runtime $O(n\log R)$; if values polynomially bounded, $O(n\log n)$.
}

\workedq{P54}{Short grid walk with value and move limit: moves right/up/diag, at most $3n/2$ moves.}{
$dp[i][j][\ell]=$ max value reaching cell $(i,j)$ in exactly $\ell$ moves. Base $dp[0][0][0]=v(0,0)$. Transition from predecessors $(i-1,j)$, $(i,j-1)$, $(i-1,j-1)$ with $\ell-1$, add $v(i,j)$. \ans $\max_{\ell\le\lfloor3n/2\rfloor}dp[n][n][\ell]$. Runtime $O(n^3)$.
}

\workedq{P55}{Rod cutting / unbounded segmentation: maximize revenue over cuts summing to $n$.}{
$r[0]=0$. For length $j=1..n$: $r[j]=\max_{1\le i\le j}(p_i+r[j-i])$. This chooses leftmost piece length $i$ and solves remainder optimally. Runtime $O(n^2)$, space $O(n)$. Store argmax to reconstruct cuts.
}

\workedq{P56}{Matrix-chain multiplication: choose parenthesization minimizing scalar multiplications.}{
Matrices $A_i$ have dimensions $p_{i-1}\times p_i$. $m[i,i]=0$. For length $\ell=2..n$: $m[i,j]=\min_{i\le k<j}(m[i,k]+m[k+1,j]+p_{i-1}p_kp_j)$. \ans $m[1,n]$. Runtime $O(n^3)$, space $O(n^2)$.
}

\workedq{P57}{LCS of two strings: subsequence, not substring.}{
$dp[i][j]=$ LCS length of prefixes $X[1..i],Y[1..j]$. Base row/col 0. If $X_i=Y_j$, $dp[i][j]=1+dp[i-1][j-1]$; else $\max(dp[i-1][j],dp[i][j-1])$. Runtime $O(nm)$. Reconstruct by walking backward.
}

\workedq{P58}{Longest increasing subsequence: standard DP and faster tails version.}{
DP: $L[j]=1+\max\{L[i]:i<j,A[i]<A[j]\}$, empty max 0; answer $\max_jL[j]$, $O(n^2)$. Tails: maintain smallest possible tail for each length; for $x$, lower\_bound first tail $\ge x$ and replace. Length of tails = LIS length, $O(n\log n)$.
}

\workedq{P59}{Longest modular increasing subsequence: increasing values and sum divisible by 3.}{
$r[j,b]=$ best length of increasing subsequence ending at $j$ with sum mod 3 equal $b$. Initialize $r[j,A_j\bmod3]=1$. For all $i<j$ with $A_i<A_j$ and all $b$: $r[j,(b+A_j)\bmod3]=\max(r[j,(b+A_j)\bmod3], r[i,b]+1)$. \ans $\max_j r[j,0]$. Runtime $O(n^2)$.
}

\workedq{P60}{0/1 knapsack: each item used at most once, capacity $W$.}{
$dp[i][w]=$ best value using first $i$ items with capacity $w$. $dp[0][w]=0$. If $weight_i>w$, skip; else $dp[i][w]=\max(dp[i-1][w], value_i+dp[i-1][w-weight_i])$. Runtime $O(nW)$ pseudo-polynomial. 1D update $w$ decreasing.
}

\workedq{P61}{3-way equal partition: split positive integers into 3 subsets each sum $W/3$.}{
If $W\not\equiv0\pmod3$, NO. Let target $S=W/3$. $dp[i][x][y]=$ possible after first $i$ items with subset1 sum $x$, subset2 sum $y$; subset3 implicit. Transition: put item $i$ in subset1 if $x\ge w_i$, subset2 if $y\ge w_i$, or subset3. \ans $dp[n][S][S]$. Runtime $O(nS^2)$.
}

\workedq{P62}{Coin-row two-player game: optimal play, take leftmost or rightmost coin.}{
$dp[i][j]=$ max value current player can guarantee from interval $i..j$. $dp[i][i]=a_i$. Recurrence $dp[i][j]=\max(a_i+\min(dp[i+2][j],dp[i+1][j-1]),\ a_j+\min(dp[i+1][j-1],dp[i][j-2]))$. Opponent chooses the branch minimizing our future. Fill by interval length.
}

\workedq{P63}{Coin change / minimum number of coins, unlimited copies.}{
$dp[0]=0$, $dp[x]=\infty$ initially. For amount $x=1..W$: $dp[x]=1+\min_{c\le x}dp[x-c]$. \ans $dp[W]$ or impossible if $\infty$. Runtime $O(W\cdot \#coins)$. For number of ways, loop coins outside and amounts increasing.
}

\workedq{P64}{Optimal BST DP with key probabilities / search frequencies.}{
$e[i,j]=$ min expected cost for keys $i..j$, $w[i,j]=$ total probability/frequency in interval. Choose root $r$: $e[i,j]=\min_{r=i..j}(e[i,r-1]+e[r+1,j]+w[i,j])$. Empty interval cost 0. Fill by increasing length. Runtime $O(n^3)$.
}

\workedq{P65}{DP answer skeleton when the problem statement is unfamiliar.}{
State meaning with units and exact interval/index convention. Base cases. Transition: enumerate the first/last decision and combine with smaller solved states. Order: dependencies first. Answer: exact table cell. Correctness: induction in fill order; optimum uses one enumerated choice and every candidate is feasible. Runtime = states $\times$ transition cost.
}

\workedq{P66}{Ski rental competitive analysis: rent price 1, buy price $B$.}{
Algorithm: rent until total rent reaches $B$, then buy. If $T\le B$, ALG=$T$=OPT. If $T>B$, ALG$\le B+B=2B=2$OPT. Lower bound for deterministic algorithms: adversary stops just before algorithm buys or continues just after; no deterministic ratio $<2$.
}

\workedq{P67}{Paging/caching LRU: cache size $k$, prove $k$-competitive.}{
Partition requests into phases with at most $k$ distinct pages; new phase starts when $(k+1)$-st distinct page appears. LRU has at most $k$ misses per phase. OPT has at least 1 miss per completed phase because phase plus next first request contains $k+1$ distinct pages. Hence LRU $\le k\cdot$OPT$+O(k)$.
}

\workedq{P68}{Weighted majority / experts: bound mistakes against best expert.}{
Weights start 1; after wrong prediction multiply expert weight by $\beta$. If algorithm makes a mistake, wrong side has weight $\ge W/2$, so $W'\le W(1+\beta)/2$. Best expert with $m^*$ mistakes has weight $\beta^{m^*}\le W_T\le n((1+\beta)/2)^M$. Solve for $M$.
}

\workedq{P69}{Secretary problem: random order, choose maximum candidate online.}{
Reject first $r=\lfloor n/e\rfloor$, remember best sample. Then accept first candidate better than all seen. Success if global best appears at position $i>r$ and best among first $i-1$ was in sample. Probability $\sum_{i=r+1}^n(1/n)(r/(i-1))\approx (r/n)\ln(n/r)\approx1/e$.
}

\workedq{P70}{Online bipartite matching: greedy and ranking guarantees.}{
Greedy matches each arriving right vertex to any free neighbor. Each ALG edge can be charged by at most two OPT edges, so ALG $\ge$ OPT/2. Ranking: pick random priority order on offline left vertices; each arrival chooses highest-priority free neighbor; expected value $(1-1/e)$OPT.
}

\workedq{P71}{Prophet inequality: threshold at half expected maximum.}{
Independent nonnegative $X_i$, known distributions. Let $M=\max_iX_i$, $z=E[M]$, threshold $\tau=z/2$. Accept first $X_i\ge\tau$. Guarantee $E[ALG]\ge z/2$. Proof uses $q=P[\forall i,X_i<\tau]$ and overshoots $(X_i-\tau)^+$; threshold contribution $\tau(1-q)$ plus overshoot contribution $q\sum_iE[(X_i-\tau)^+]$.
}

\workedq{P72}{Two Sum / pair with target: hash previous values.}{
Scan left to right. Before processing $i$, map contains values from indices $<i$. Need $target-A[i]$ in map; if yes return pair. Else insert $A[i]\mapsto i$. Correct by invariant: when second element of any solution is scanned, first is already stored. Expected $O(n)$.
}

\workedq{P73}{Valid parentheses / bracket matching: stack invariant.}{
Scan chars. Push opening brackets. On closing bracket, stack top must be matching opening; otherwise reject. End accept iff stack empty. Invariant: stack contains unmatched openings in order. Runtime $O(n)$.
}

\workedq{P74}{Number of islands / connected components in a grid.}{
For every unvisited land cell, increment count and BFS/DFS over 4-neighbors marking reachable land. Each land cell visited once and each adjacency checked constant times. Runtime $O(nm)$, space $O(nm)$ worst-case queue/recursion.
}

\workedq{P75}{Redundant connection / first edge creating an undirected cycle.}{
Initialize DSU. For each edge $(u,v)$ in order: if find$(u)=$find$(v)$, this edge closes a cycle, return it; otherwise union. Correct because DSU tracks connectivity in graph built by previous edges. Runtime $O(E\alpha(V))$.
}

\workedq{P76}{Top-$k$ frequent elements: frequency map plus heap/buckets.}{
Count occurrences with HashMap. Heap version: push $(freq,key)$ into max-heap and pop $k$, $O(n+m\log m)$. Bucket version: array buckets by frequency $0..n$, append keys to bucket[f], scan buckets downward until $k$ keys, $O(n+m)$.
}

\workedq{P77}{Sliding window with all elements distinct: longest substring/subarray no repeats.}{
Maintain left pointer $l$ and map last position. For each right $r$, if $A[r]$ last seen at $p\ge l$, set $l=p+1$. Update last and answer $r-l+1$. Invariant: window $[l,r]$ has no duplicates and is the longest valid window ending at $r$. Runtime $O(n)$ expected.
}

\workedq{P78}{K sorted lists / merge streams with priority queue.}{
Push first element of each list with list id into min-heap. Repeatedly pop smallest, output it, then push next element from same list. Heap size at most $k$, each element pushed/popped once. Runtime $O(N\log k)$, space $O(k)$.
}



\workedq{P79}{Maximum subarray: D\&C crossing case or Kadane linear DP.}{
D\&C: best subarray is in left half, right half, or crosses mid. Crossing = best suffix of left + best prefix of right, found by scans. $T(n)=2T(n/2)+O(n)=O(n\log n)$. Kadane: $\text{bestEnd}[i]=\max(A_i,\text{bestEnd}[i-1]+A_i)$, answer $\max_i \text{bestEnd}[i]$, O(n).
}

\workedq{P80}{Count inversions by merge sort.}{
Split array, count inversions in left/right recursively. During merge, when right element $R[j]$ is copied before remaining left elements $L[i..]$, add $|L|-i+1$ inversions. Merge still sorted. Runtime $T(n)=2T(n/2)+O(n)=O(n\log n)$.
}

\workedq{P81}{Merge sort correctness and stability.}{
Induct on length. Recursive calls sort both halves. Merge invariant: output prefix is sorted and contains the smallest already-consumed elements from both halves. If ties choose left element first, stable order is preserved. Runtime $O(n\log n)$, space $O(n)$.
}

\workedq{P82}{Quickselect / randomized selection: kth smallest expected linear.}{
Partition around random pivot. If pivot rank $r=k$ return; if $k<r$ recurse left; else recurse right with $k-r$. Expected recurrence $E[T(n)]\le E[T(\text{random side})]+O(n)=O(n)$; good pivots shrink to at most $3n/4$ with constant probability.
}

\workedq{P83}{BFS unweighted shortest paths: distance labels are exact.}{
Queue processes vertices by nondecreasing distance. When discovering $v$ from $u$, set $d[v]=d[u]+1$. Any shorter path to $v$ has predecessor at smaller distance, which would be processed earlier and would have discovered $v$ earlier. Hence BFS distances are shortest edge counts.
}

\workedq{P84}{Multi-source BFS: nearest source / simultaneous spread.}{
Initialize queue with all sources at distance 0. Then run ordinary BFS. Equivalent to adding super-source connected to all sources by 0/1 edges. First time a vertex is popped gives minimum distance to any source. Runtime $O(V+E)$.
}

\workedq{P85}{DFS cycle detection and topological order.}{
Use colors WHITE/GRAY/BLACK. In directed graph, edge to GRAY ancestor is a back edge and proves a cycle. If no back edges, graph is DAG. Decreasing finish times give topological order because for every edge $u\to v$, DFS implies $f[u]>f[v]$.
}

\workedq{P86}{Undirected cycle detection with DFS/DSU.}{
DFS: when exploring edge $(u,v)$, if $v$ already visited and $v\ne parent[u]$, cycle exists. DSU online: for each undirected edge, if endpoints already have same representative, edge closes a cycle; otherwise union. Runtime $O(V+E)$ or $O(E\alpha V)$.
}

\workedq{P87}{SCC condensation graph: reduce directed graph to a DAG.}{
Compute SCC ids. Build condensation nodes = components, edge $C\to D$ if original edge crosses components. It is a DAG: a directed cycle among components would imply all those components are mutually reachable and should be one SCC. Use topo DP on condensation for reachability/path tasks.
}

\workedq{P88}{Kruskal trace with equal weights: when ties change MST but not weight.}{
Sort edges by nondecreasing weight; scan. Accept edge iff endpoints currently in different DSU components. With equal weights, different tie orders may produce different MSTs. Cut property still safe because accepted edge is lightest crossing some component cut.
}

\workedq{P89}{Second-best MST / replacement edge idea.}{
Compute MST $T$. For each non-tree edge $e=(u,v)$, adding $e$ creates cycle. To get a spanning tree, remove the maximum-weight edge $f$ on path $u\leadsto v$ in $T$. Candidate weight $w(T)+w(e)-w(f)$. Minimum candidate strictly larger than $w(T)$ is second-best. Naive $O(EV)$.
}

\workedq{P90}{Max-flow optimality certificate and min-cut extraction.}{
After Ford-Fulkerson/Edmonds-Karp terminates, let $S$ be vertices reachable from $s$ in residual graph. No residual edge leaves $S$, so crossing edges $S\to T$ are saturated and $T\to S$ carry zero. Thus $|f|=c(S,T)$. Since every flow $\le$ every cut, both are optimal.
}

\workedq{P91}{Minimum $s$-$t$ cut from max-flow result.}{
Run maxflow. In final residual graph, DFS/BFS from $s$ using positive residual capacity. Let reachable set be $S$. Return cut edges $(u,v)$ with $u\in S$, $v\notin S$. Capacity equals maxflow value by max-flow min-cut theorem.
}

\workedq{P92}{Project selection / maximum closure as min-cut.}{
Profit $p_v$ can be positive/negative; dependencies $u$ chosen $\Rightarrow v$ chosen. Add $s\to v$ cap $p_v$ if positive, $v\to t$ cap $-p_v$ if negative, dependency edge $u\to v$ cap INF. Min cut chooses profitable closed set; max profit = total positive profit $-$ mincut.
}

\workedq{P93}{Interval partitioning / minimum number of rooms.}{
Sort intervals by start time. Maintain min-heap of room finish times. For interval $(s,f)$, if min finish $\le s$, reuse that room by pop then push $f$; else allocate new room and push $f$. Max heap size = max overlap = optimal number of rooms. Runtime $O(n\log n)$.
}

\workedq{P94}{Sweep line: maximum overlap of intervals.}{
Create events $(start,+1),(end,-1)$; sort by coordinate. For half-open intervals $[s,e)$ process end before start at same coordinate. Running sum gives active count; maximum is max overlap/min rooms. Runtime $O(n\log n)$.
}

\workedq{P95}{LRU cache data structure in $O(1)$ expected time.}{
Use HashMap key$\to$node in doubly linked list ordered by recency. Get: if key present, move node to front. Put: update/move if present; else insert front and evict tail if over capacity. HashMap gives lookup, list gives delete/insert known node in $O(1)$.
}

\workedq{P96}{LFU cache data structure idea in $O(1)$ expected time.}{
Use key$\to$(value,freq,node), freq$\to$ doubly linked list of keys with that freq, and minFreq. Get/update: remove key from old freq list, increment freq, insert into new freq list; update minFreq if old list empty. Evict from minFreq list using LRU tie-break.
}

\workedq{P97}{LRU vs LFU MC trap: neither dominates pointwise.}{
To disprove ``LRU always at most LFU misses'' or reverse, give one sequence where LFU keeps an old frequent page but misses recent working-set pages, and another where LRU evicts a frequently reused page after a scan. Course fact: LRU is $k$-competitive; LFU does not have that pointwise guarantee.
}

\workedq{P98}{Group anagrams: canonical key in hash map.}{
For each word, compute key = sorted characters or 26-count vector. Push word into HashMap[key]. Correct because anagrams have identical character multiset iff keys equal. Runtime sorting-key $O(\sum L_i\log L_i)$ or count-key $O(\sum L_i)$ for fixed alphabet.
}

\workedq{P99}{Valid anagram / frequency equality.}{
If lengths differ, false. Count characters of first string, subtract counts for second. Reject if any count negative / final nonzero. Correct because two strings are anagrams iff every character multiplicity agrees. Runtime $O(n)$ for fixed alphabet or expected $O(n)$ with hash map.
}

\workedq{P100}{House Robber path DP: choose non-adjacent items.}{
$dp[i]=$ best value using first $i$ houses. Base $dp[0]=0$, $dp[1]=a_1$. Recurrence $dp[i]=\max(dp[i-1],a_i+dp[i-2])$. Correct: optimum either skips house $i$ or takes it and cannot take $i-1$. Runtime $O(n)$, space $O(1)$ with two variables.
}

\workedq{P101}{Tree House Robber / independent set on tree.}{
Postorder returns pair $(take,skip)$. $take(u)=w_u+\sum_{child}skip(child)$. $skip(u)=\sum_{child}\max(take(child),skip(child))$. \ans $\max(take(root),skip(root))$. This is tree DP: parent-child adjacency is the only conflict.
}

\workedq{P102}{Word Break DP: segment string using dictionary.}{
$dp[i]=$ whether prefix $s[0..i)$ can be segmented. $dp[0]=true$. $dp[i]=\bigvee_{j<i}(dp[j]\land s[j..i]\in Dict)$. \ans $dp[n]$. Runtime $O(n^2)$ dictionary checks, plus substring cost depending implementation.
}

\workedq{P103}{Edit distance DP: insert/delete/replace.}{
$dp[i][j]=$ min edits from $A[1..i]$ to $B[1..j]$. Base $dp[i][0]=i$, $dp[0][j]=j$. If chars equal, $dp[i][j]=dp[i-1][j-1]$; else $1+\min(dp[i-1][j],dp[i][j-1],dp[i-1][j-1])$. Runtime $O(nm)$.
}

\workedq{P104}{Shortest path with at most $K$ edges / flights with $K$ stops.}{
Layered DP/Bellman-Ford. $dp[\ell][v]=$ shortest distance to $v$ using at most/exactly $\ell$ edges. Initialize $dp[0][s]=0$. Relax all edges for $\ell=1..K$: $dp[\ell][v]=\min(dp[\ell-1][v],\min_{u\to v}dp[\ell-1][u]+w(u,v))$. Runtime $O(KE)$.
}

\workedq{P105}{Shortest path with one discounted/free edge.}{
State $(v,used)$ where $used\in\{0,1\}$. For each edge $u\to v$ weight $w$: normal transition $(u,used)\to(v,used)$ cost $w$; if $used=0$, discount transition $(u,0)\to(v,1)$ cost discounted weight. Run Dijkstra if all costs nonnegative. Runtime $O(E\log V)$ up to factor 2.
}

\workedq{P106}{Kth smallest in BST with subtree sizes.}{
At node $x$, let $L=size(x.left)$. If $k=L+1$, return $x$. If $k\le L$, go left. Else go right with $k-L-1$. Runtime $O(h)$. This is rank/select dual to interval counting.
}

\workedq{P107}{Validate BST recursively with open intervals.}{
Function valid(x,lo,hi): if null true; require $lo < x.key < hi$ (or adjust for duplicate convention); recurse left with $(lo,x.key)$ and right with $(x.key,hi)$. Correct because local parent checks alone do not catch deeper violations. Runtime $O(n)$.
}

\workedq{P108}{Topological DP for counting paths in a DAG.}{
Topo sort. Initialize $ways[s]=1$. For vertices in topo order, for each edge $u\to v$, add $ways[u]$ to $ways[v]$. Correct because all paths to $v$ end with some predecessor processed earlier. Runtime $O(V+E)$.
}
