\proofbanktitle{Exam Writeup Checklist}
\textbf{Algorithm:} input/output, data structs, return value.
\textbf{Correctness:} complete invariant/induction/exchange proof, not idea.
\textbf{DP:} State, Base, Transition, Order, Answer, Runtime, Space.
\textbf{Bounds:} define $n,V,E,W$; justify loops.
\textbf{Conventions:} $<$ vs $\le$, inclusive/exclusive, 0/1-index, reachable vs global negative cycle.

\proofbanktitle{Proof Bank: Generic Templates}
\proofitem{Loop inv}{Init true before loop. Maint: one step preserves. Term: invariant + stop condition $\Rightarrow$ answer.}

\proofitem{Induction/substitution}{Claim upper/lower bound with constants. State IH for all smaller inputs. Substitute IH. Collect terms. Pick constants so inequality holds. Base cases: choose $c$ large enough for all $n<n_0$.}

\proofitem{Substitution trap}{If $T(n)\le c\sum_i(a_in)^p+bn$ and $\sum_i a_i^p=1$, then $cn^p+bn$ FAILS. No slack absorbs $+bn$. Strengthen: $T(n)\le cn^p-dn^q$ with lower-order subtraction.}

\proofitem{Midterm recurrence}{$T(n)=T(4n/5)+T(3n/5)+an$. Guess $\Theta(n^2)$ because $(4/5)^2+(3/5)^2=1$. Lower: $T(n)\ge T(4n/5)+T(3n/5)\ge c((4n/5)^2+(3n/5)^2)=cn^2$. Upper must be strengthened: assume $T(m)\le cm^2-dm$ for all $m<n$. Then $T(n)\le c(4n/5)^2-d(4n/5)+c(3n/5)^2-d(3n/5)+an=cn^2-7dn/5+an\le cn^2-dn$ iff $d\ge5a/2$. Choose $d$ so, then choose $c$ large for base cases.}

\proofitem{D\&C}{Induct on size. Recursive calls solve smaller problems. Combine covers all optimum cases and constructs/chooses correct result.}

\proofitem{DP}{State meaning. Base correct. Induct in fill order. Transition enumerates all valid first/last choices; dependencies already optimal. Optimal solution uses one enumerated choice; every candidate feasible. Therefore $dp[s]$ optimum. Write answer $=dp[\dots]$.}

\proofitem{Parametric search}{Guess answer $M$. Build feasible$(M)$. Prove monotone: if $M$ works then any larger $M$ works (minimization). Binary search smallest feasible $M$. Correctness = binary search invariant + feasible predicate exact.}

\proofitem{Greedy}{Safe choice: for optimum $O$ containing current partial solution, if chosen $e\notin O$, exchange $f\in O$ with $e$ without worsening. Then induct.}

\proofbanktitle{Proof Bank: Sorting / Hash / DS}
\proofitem{BST inorder}{Induct on subtree. Left keys $<$ root, right keys $\ge$ root. IH sorts left/right. Left,root,right sorted and complete.}

\proofitem{Rank/interval}{Store $size=1+size(L)+size(R)$. Rank follows one search path; when going right add $size(L)+1$. Thus $O(h)$. $[a,b]=rank_{\le}(b)-rank_{<}(a)$.}

\proofitem{Heapify}{Children are heaps. Swap with larger child if violated; parent fixed. Only swapped child may violate; recurse to leaf. Cost height $O(\log n)$.}

\proofitem{BuildHeap}{Leaves are heaps. Bottom-up: children already heaps, so Heapify fixes subtree. Runtime: nodes height $h\le n/2^{h+1}$, total $\sum_h O(h)n/2\textasciicircum h=O(n)$.}

\proofitem{Sorting lower bound}{Comparison sort decision tree needs $\ge n!$ leaves. Worst-case comparisons $\ge$ height $\ge\log_2(n!)=\Omega(n\log n)$.}

\proofitem{Quicksort exp}{Pair compared at most once. $X_{ij}=1$ if ranks $i,j$ compared. Compare iff first pivot among ranks $i..j$ is $i$ or $j$, so $P(X_{ij})=2/(j-i+1)$. Sum $O(n\log n)$.}

\proofitem{Hashing}{Single key: for key $x$, $X_y=1[h(y)=h(x)]$. Universal hashing gives $E[X_y]\le1/m$, so expected chain collisions $\le(n-1)/m=\Theta(\alpha)$ and search $O(1+\alpha)$. Total collisions: $X=\sum_{i<j}X_{ij}$, $X_{ij}=1[h(x_i)=h(x_j)]$, hence $E[X]\le{n\choose2}/m=n(n-1)/(2m)=\Theta(n2/m)$. $k$-collision: $E[\#k\text{-collisions}]\le {n\choose k}/m^{k-1}$. Markov: $P[X\ge1]\le E[X]$.}
\proofitem{Hash mini}{Length-$\ell$ keys: pair collision prob $1/m^\ell$, so $E[\#collisions]={n\choose2}/m^\ell$; if $m^\ell\ge{n\choose2}$ then $P[\exists collision]\le1$. Random $q$-color vertices: edge mono prob $1/q$, so $E[\#mono]=|E|/q$.}
\proofitem{DSU}{Union by rank attaches smaller rank under larger. Rank increases only on equal-rank merge. Path compression preserves reps and shortens future paths. With both: $m$ ops in $O(m\alpha(n))$.}

\proofbanktitle{Proof Bank: Graph Traversal}
\proofitem{BFS}{Queue processed in nondecreasing distance. Discover $v$ from $u$: $d[v]=d[u]+1$. Shorter path would come from smaller-distance vertex processed earlier, so $v$ would already be discovered.}

\proofitem{DFS/topo}{Directed cycle iff DFS has back edge to GRAY ancestor. DAG has no back edges. For every edge $u\to v$, $f[u]>f[v]$; decreasing finish time is topo order.}

\proofitem{DFS edge facts}{Parenthesis theorem: finish intervals disjoint or nested; nested iff descendant. White-path theorem: $v$ descendant of $u$ iff when $u$ discovered, there is a white path $u\leadsto v$.}

\proofitem{SCC/Kosaraju}{Condensation graph is DAG. First DFS finish order puts source SCCs late. In reversed graph, DFS from highest finish cannot leave its SCC to an unvisited SCC. Each second-pass DFS outputs one SCC.}

\proofitem{SCC reachability pairs}{$u\leadsto v$ and $v\leadsto u$ iff $comp[u]=comp[v]$. For forbidden pair $(u,v)$ reject iff same component. Correct because SCC is maximal mutual reachability. Runtime $O(V+E)$.}

\proofitem{DAG SP/LP}{Topo order processes every predecessor before vertex. When processing $u$, $dist[u]$ already final over all paths ending at $u$; relaxing outgoing edges propagates optimum once. Runtime $\Theta(V+E)$.}

\proofbanktitle{Proof Bank: MST / Shortest Paths / Flow}
\proofitem{Cut property}{Let $e$ be light edge crossing cut respecting $A$. Take MST $T\supseteq A$ not using $e$. Add $e$ creates cycle with another crossing edge $f$. Since $w(e)\le w(f)$, $T-f+e$ is MST containing $A\cup\{e\}$.}

\proofitem{Cycle property}{If MST $T$ contains unique heaviest edge $e$ on cycle, remove $e$; cycle has another edge $f$ reconnecting cut with $w(f)<w(e)$. Then $T-e+f$ cheaper, contradiction.}

\proofitem{Kruskal}{DSU components define cut respecting chosen edges. Next accepted edge is lightest crossing some component cut. Cut property says safe. Induct until $|V|-1$ edges.}

\proofitem{Prim}{$S=$ vertices in tree. Heap chooses lightest edge crossing $(S,V\setminus S)$. Cut property says safe. Add one vertex; repeat.}

\proofitem{Dijkstra}{$S=$ finalized. Pick outside $x$ with min tentative dist. Any alternative path first crosses from $S$ to outside at $(u,v)$. Since $w\ge0$, path length $\ge d[u]+w(u,v)\ge d[v]\ge d[x]$. So $d[x]$ final.}

\proofitem{Bellman-Ford}{After $i$ passes, all shortest paths using $\le i$ edges are correct. Induct by last edge relaxation. If no negative cycle, shortest simple path has $\le V-1$ edges. If pass $V$ improves, reachable negative cycle.}

\proofitem{Max-flow}{If no augmenting path exists, let S be vertices reachable from s
in residual graph. For every edge $u\in S$,$v\notin S$: $cf(u,v)=0$, therefore $f(u,v)=c(u,v)$.
For every edge $u\notin S$,$v\in S$:
$cf(v,u)=0$, therefore $f(u,v)=0$.

Hence net flow crossing cut equals cut capacity:
$|f| = c(S,T)$.

Since every flow $\le$ every cut,
flow and cut are both optimal.}

\proofitem{Flow cert}{Any flow value $\le$ any cut capacity. Flow value $F$ + cut capacity $F$ proves both optimal.}

\proofitem{Bipartite matching flow}{Caps 1 force each left/right vertex used at most once. Integral capacities give integral max flow. Unit flow through $l\to r$ is matched pair; every matching gives same-size flow.}

\proofitem{Cap assignment flow}{Build $s\to$ people cap 1, person$\to$ acceptable item cap 1, item$\to t$ cap capacity. Feasible iff maxflow = \#people. Integral flow = assignment; assignment = integral flow.}

\proofitem{SP transform}{Multiplying every edge weight by $a>0$ preserves all shortest paths. Adding $a$ to every edge does not: paths with different number of edges get different extra cost.}

\proofitem{Resource SP}{State graph $(v,b)$ where $b=$ used budget / remaining battery. Add transition only if resource feasible. Run BFS/Dijkstra depending on edge costs. Size $O(VB+EB)$. Correct since every feasible walk corresponds to one state path and vice versa.}
\proofitem{2D sorted matrix}{Rows/cols sorted. Start top-right. If $A[i,j]>x$, move left; if $A[i,j]<x$, move down. Each step deletes one impossible row/col, so $O(n+m)$, $O(1)$.}
\proofbanktitle{Proof Bank: Trees / Exam Reductions}
\proofitem{Tree DP}{Postorder. Return height/best info from children. Best at $u$ is max(best in left, best in right, best crossing/using $u$). Induct on subtree: any optimal object either lies fully in one child or uses $u$.}

\proofitem{Tree diameter}{Let $h(u)=1+\max(h_L,h_R)$, empty height $-1$. $diam(u)=\max(diam_L,diam_R,h_L+h_R+2)$. Path either inside one child or crosses through $u$.}
\proofitem{Root path}{Longest simple path forced to pass through root: $h(root.left)+h(root.right)+2$ with empty height $-1$. Compute heights postorder in $O(n)$.}
\proofitem{Tree rebuild}{Inorder+postorder: root=last(post); split inorder at root; subtree sizes split postorder; recurse. Preorder+inorder: root=first(pre). Pre+post alone not unique.}

\proofitem{Unicyclic spanning trees}{Connected graph with $|E|=|V|$ has exactly one cycle $C$. Every spanning tree = delete exactly one edge of $C$. Deleting non-cycle edge disconnects; deleting cycle edge breaks only cycle. Hence \#spanning trees $=|C|$.}


\proofbanktitle{Proof Bank: Online / Random}

\proofitem{LRU paging}{Cache size $k$. Phase has $\le k$ distinct pages; new phase begins at request of $(k+1)$-st distinct. LRU: $\le k$ misses/phase, because requested page survives until $k$ other distinct pages requested. OPT: $\ge1$ miss/completed phase, because phase + first request of next phase contains $k+1$ distinct pages but cache holds $k$. Thus LRU $\le k\cdot$OPT$+O(k)$; LRU is $k$-competitive.}


\subsectiontitle{Verbose Proof Micro-Stencils}
{
\microq{Correctness opening}{Start with a precise claim: after the algorithm terminates, the returned object is feasible and has objective value no worse than any feasible object. Then prove feasibility and optimality separately.}
\microq{Induction base habit}{Say exactly which subproblems are smallest and why their table/return value is correct. Do not write ``trivial'' unless the value is actually shown, e.g. dp[0]=0 or empty subtree height=-1.}
\microq{Induction step habit}{Assume all strict subproblems used by the transition are correct. Show every transition candidate is feasible, and the optimum's first/last choice appears among the candidates.}
\microq{Exchange proof form}{Let O be an optimum agreeing with greedy for as long as possible. If greedy choice g differs from O's next choice o, replace o by g; prove feasibility remains and weight/count is not worse. Contradiction to maximal agreement.}
\microq{Greedy lower-bound proof}{To prove greedy is optimal, also prove a matching lower bound: any solution needs at least X objects/cost. Then show greedy achieves X. Common in interval partitioning and stabbing.}
\microq{Cut proof exactness}{For MST safe edge proofs, the cut must respect already chosen edges A: no edge in A crosses it. Then swapping along the cycle cannot remove an edge of A.}
\microq{Cycle proof exactness}{For MST cycle proofs, removing the heavy edge disconnects T into a cut. The replacement edge on the cycle crosses the same cut, reconnects T, and has smaller/equal weight.}
\microq{Reduction correctness}{Prove both directions: any original solution maps to a feasible constructed solution of same value, and any constructed solution maps back to a valid original solution of same value. Then optimum values equal.}
\microq{Flow reduction proof}{Integral capacities imply an integral max flow. Unit flow through a gadget means the corresponding discrete choice is selected; capacity constraints exactly encode usage limits.}
\microq{Matching proof}{Flow->matching: cap 1 on s->L and R->t gives at most one matched edge per vertex. Matching->flow: send one unit along s-l-r-t for each matched pair.}
\microq{Min-cut certificate}{After max-flow, BFS in residual graph from s gives S. Edges S->T are saturated and T->S carry zero flow, so value crossing cut equals cut capacity.}
\microq{Shortest path reduction}{When adding state dimensions, prove bijection-ish: every feasible original walk corresponds to a path in the state graph, and every state path decodes to a feasible walk.}
\microq{Dijkstra finality}{The key sentence: when x is extracted, any alternative path to x must first leave the finalized set by some edge (u,v); nonnegative weights make that path at least dist[v] >= dist[x].}
\microq{Bellman-Ford invariant}{After i full passes, dist[v] is at most the length of the shortest s->v path using at most i edges, and no smaller value than any such path is needed. Last-edge relaxation proves it.}
\microq{Negative-cycle proof}{If pass V still relaxes an edge reachable from s, the improving walk has V edges, so some vertex repeats; the repeated cycle must have negative total weight.}
\microq{BFS shortest proof}{Use contradiction: let v be first vertex discovered with nonshortest distance. Its predecessor on a shortest path has smaller distance and was processed earlier, so it would have discovered v correctly.}
\microq{DFS cycle proof}{Directed cycle -> during DFS, take the first vertex on the cycle discovered; following the cycle reaches a gray ancestor edge before it finishes. Gray edge -> ancestor path plus edge forms a cycle.}
\microq{Topo proof}{In a DAG DFS has no back edge. For every edge u->v, v finishes before u whether discovered from u or already black. Therefore decreasing finish time respects every edge.}
\microq{SCC proof}{Condensation is a DAG. The second DFS in reverse graph starts from a source of remaining reverse condensation, equivalently a sink/source by finish order, so it cannot escape its SCC.}
\microq{Binary search invariant}{Maintain answer in [l,r]. Mid test discards only values impossible by monotonicity. Termination l=r gives the unique smallest feasible or largest feasible depending on direction.}
\microq{Feasibility proof}{For parametric search, prove feasible(M) is exact, not just sufficient. Usually: greedy uses the minimum possible resources for this M, so if greedy fails no valid solution exists.}
\microq{Divide-farm proof}{For fixed M, greedy makes each segment as long as possible. After t segments greedy covers at least as much prefix as any t-segment valid partition; hence greedy minimizes number of segments.}
\microq{Interval scheduling proof}{Let g be earliest finishing interval. For any optimum, replace its first interval by g. Since g finishes no later, all later intervals remain compatible; size unchanged. Induct on suffix.}
\microq{Weighted interval DP proof}{Optimal solution for first i intervals either skips i or takes i. If it takes i, all remaining chosen intervals lie among first p(i). These two cases are exhaustive and disjoint.}
\microq{Travel-pass DP proof}{In an optimum covering first uncovered day d[i], the bought pass has one of the allowed lengths L and covers exactly until d[i]+L-1. The remaining subproblem starts at next(i,L).}
\microq{Knapsack proof}{For item i, optimum either excludes it, giving dp[i-1][w], or includes it, forcing capacity w-weight[i] for previous items. No third possibility exists.}
\microq{LCS proof}{For prefixes i,j: if last chars match, some optimum can use them and reduce to i-1,j-1. If they differ, at least one last char is not used, giving max of the two smaller prefixes.}
\microq{Matrix-chain proof}{The final multiplication splits the product at some k. Costs before that split are independent subchains, so optimality follows by trying every k and using optimal subsolutions.}
\microq{Rod-cutting proof}{Look at the leftmost piece length i in an optimal cutting. The remainder length n-i must itself be cut optimally, else replace it and improve total revenue.}
\microq{Optimal BST proof}{Choosing root r increases depth of every key in left/right subtrees by 1, adding total interval probability w[i,j]. Subtrees are independent optimal BST subproblems.}
\microq{Tree DP proof}{Any optimal object in subtree u either lies completely in one child subtree or uses u / crosses between children. The recurrence takes the max over exactly these cases.}
\microq{Tree reconstruction proof}{Root position in inorder uniquely separates left and right key sets. Subtree sizes then uniquely split preorder/postorder blocks, so recursion reconstructs the only tree.}
\microq{Heapify proof}{Before heapify, both child subtrees are heaps. Swapping with the larger child fixes the parent; the only possible violation moves down one level. Eventually a leaf is reached.}
\microq{Build-heap proof}{Process internal nodes from bottom to top. When heapify(i) runs, both children already root heaps, so it makes subtree i a heap. At the root the whole array is a heap.}
\microq{Insertion sort proof}{Before iteration j, prefix A[0..j) is sorted and contains the original prefix elements. Inner loop shifts larger elements right; inserting key restores sorted prefix A[0..j+1).}
\microq{Merge proof}{Loop invariant: output contains the smallest consumed elements from both sorted halves, in sorted order. Taking the smaller head preserves sortedness; copying leftovers completes all elements.}
\microq{Quicksort proof}{Partition puts pivot in final rank: all left <= pivot, all right > pivot. Recursive calls sort the two sides; concatenating left,pivot,right is sorted and complete.}
\microq{Counting-sort proof}{Prefix count C[x] equals number of input keys <=x. Scanning input right-to-left places each key at the last free position of its value, giving sorted stable output.}
\microq{Hash expectation proof}{Define one indicator per bad pair/event. Compute probability for a fixed pair, then sum expectations. Independence between different pairs is not needed for the expectation.}
\microq{Quicksort expectation proof}{Elements of ranks i,j are compared iff the first pivot chosen among ranks i..j is one of them. That probability is 2/(j-i+1); summing pairs gives O(n log n).}
\microq{Lower-bound proof}{For comparison sorting, leaves of decision tree must distinguish all n! input orders. A binary tree of height h has at most 2\textasciicircum h leaves, so h>=log(n!).}
\microq{Online upper proof}{Partition the request sequence into phases. Prove ALG cost per phase <=A and OPT cost per phase >=B, then sum phases and absorb boundary phases into constant c.}
\microq{Online lower proof}{Adversary observes the deterministic algorithm's action and chooses when to stop/continue. Compare on the same final sequence against offline OPT.}
\microq{Ski-rental proof}{If T<=B, rent-then-buy pays T=OPT. If T>B, it pays at most B rent plus B buy, while OPT pays B; ratio <=2.}
\microq{LRU phase proof}{In one phase there are at most k distinct pages, so after a page is requested, LRU will not evict it before k other distinct pages appear, which would start a new phase.}
\microq{Experts potential proof}{Track total weight W. Algorithm mistake means wrong side has at least half W, so W shrinks by factor <=(1+beta)/2. Best expert gives lower bound beta\textasciicircum m*.}
\microq{Prophet threshold proof}{Separate value into threshold part tau*P[accept] and overshoot part. If all earlier variables miss threshold, each positive overshoot has chance to be captured; combine with tau=z/2.}
\microq{Secretary proof}{Condition on best arriving at position i>r. Success iff best among first i-1 positions is in sample. Probability sum over i: (1/n)*(r/(i-1)).}
\microq{Runtime proof}{Count objects actually processed: each vertex enqueued once, each edge scanned once, each DP state filled once, each heap operation O(log size). State dimensions after reductions.}
\microq{Off-by-one proof}{State intervals explicitly. Half-open [l,r) usually makes size r-l, empty when l=r, and split [l,m),[m,r). Closed [i,j] needs base i>j or i=j.}
\microq{Final answer habit}{After proof, write the exact returned value: dp[n], dp[1], dist[t], maxflow value, selected edges, or reconstructed choices. Missing this often loses points.}

\subsectiontitle{Hash Tables}

\textbf{Idea:} bucket $h(key)\bmod m$. \textbf{Collisions:} chaining / probing. \textbf{Load factor} $\alpha=n/m$.

Hash ops: insert/delete expected O(1), search expected O(1+alpha), all worst O(n); m=Theta(n) => alpha=Theta(1).
\textbf{Expected total collisions:}
$X=\sum_{i<j}X_{ij}$, where $X_{ij}=1[h(x_i)=h(x_j)]$.
Universal hashing gives P[Xij=1]<=1/m, so E[X]=sum E[Xij] <= C(n,2)/m = n(n-1)/(2m)=Theta(n2/m).
For one key: $\mathbb E[\#\text{collisions}]\le(n-1)/m=\Theta(\alpha)$, hence expected search $O(1+\alpha)$.
\textbf{Traps:} HashMap/HashSet $O(1)$ is expected/amortized, not worst-case.
Bad hashing / all collisions $\Rightarrow O(n)$.

\textbf{k-collision stencil:} $X_S=1$ if fixed $k$-set collides. $P[X_S=1]\le1/m^{k-1}$. $E[X]=\sum_{|S|=k}E[X_S]\le {n\choose k}/m^{k-1}$. If need existence prob: $P[X\ge1]\le E[X]$.

fn counting_sort(a:&Vec<usize>, k:usize)->Vec<usize>{
  let mut c=vec![0;k+1]; let mut out=vec![0;a.len()];
  for &x in a { c[x]+=1; }
  for i in 1..=k { c[i]+=c[i-1]; }

fn max_heapify(a:&mut Vec<i32>, i:usize, heap_size:usize){
  let (l,r)=(2*i,2*i+1); let mut largest=i;
  if l<heap_size && a[l]>a[largest] { largest=l; }
  if r<heap_size && a[r]>a[largest] { largest=r; }

}

\sectiontitle{Dynamic Programming}
\subsectiontitle{Rod Cutting}
revenue$[\ell]=$ best revenue length $\ell$, $\text{first}[\ell]=$ first cut.
Rod cutting: revenue[0]=0; revenue[l]=max over 1<=c<=l of price[c]+revenue[l-c].
\begin{lstlisting}[aboveskip=0pt,belowskip=0pt]
fn rod_cut(price:Vec<i32>, n:usize){
  revenue=vec![0;n+1]; first=vec![0;n+1];
  for length in 1..=n {revenue[length]=-INF;
    for cut in 1..=length {
      candidate=price[cut]+revenue[length-cut];
      if candidate>revenue[length] {
        revenue[length]=candidate; first[length]=cut
      }
    }
  }
  (revenue,first)
}
\end{lstlisting}
Reconstruct: output `first[length]`, set `length-=first[length]`. Runtime $\Theta(n^2)$.

\subsectiontitle{Matrix Chain Multiplication}
Matrix $i$: $p_i\times p_{i+1}$. $dp[i][j]=$ min cost for $A_i\cdots A_j$.
MCM recurrence: dp[i][i]=0; dp[i][j]=min over i<=k<j of dp[i][k]+dp[k+1][j]+p[i]*p[k+1]*p[j+1].
\begin{lstlisting}[aboveskip=0pt,belowskip=0pt]
fn matrix_chain(p:Vec<usize>){     // optimal parenthesization
  n=p.len()-1; dp=vec![vec![0;n];n]; split=vec![vec![0;n];n];
  for len in 2..=n {for i in 0..=n-len {
    j=i+len-1; dp[i][j]=INF;
    for k in i..j {
      cost=dp[i][k]+dp[k+1][j]+p[i]*p[k+1]*p[j+1];
      if cost<dp[i][j] {dp[i][j]=cost; split[i][j]=k}
    }
  }}
  (dp,split)
}
\end{lstlisting}
Increasing chain length. $\Theta(n^2)$ states $\times\Theta(n)$ splits $=\Theta(n^3)$.

\subsectiontitle{Optimal BST}
Sorted keys $K_i$ with successful probs $p_i$; optional unsuccessful $q_i$. $e[i][j]=$ min expected cost for subtree $i..j$, $w[i][j]=$ prob mass, $root[i][j]=$ best root.
OBST: e[i][i-1]=0, w[i][i-1]=q[i-1]; e[i][j]=min over root r in [i,j] of e[i][r-1]+e[r+1][j]+w[i][j].
\begin{lstlisting}[aboveskip=0pt,belowskip=0pt]
fn optimal_bst(p,q,n){
  for i in 1..=n+1 {e[i][i-1]=0; w[i][i-1]=q[i-1]}
  for len in 1..=n {for i in 1..=n-len+1 {
    j=i+len-1; e[i][j]=INF; w[i][j]=w[i][j-1]+p[j]+q[j];
    for r in i..=j {
      cost=e[i][r-1]+e[r+1][j]+w[i][j];
      if cost<e[i][j] {e[i][j]=cost; root[i][j]=r}
    }
  }}
  (e,root)
}
\end{lstlisting}
Runtime Theta(n\textasciicircum3), space Theta(n\textasciicircum 2).

\subsectiontitle{Knapsack 0/1}
$dp[i][c]=$ best value using first $i$ items, capacity $c$.
0/1 knapsack: dp[i][c]=max(skip dp[i-1][c], take v[i-1]+dp[i-1][c-w[i-1]]) if w[i-1]<=c.
\begin{lstlisting}[aboveskip=0pt,belowskip=0pt]
fn knapsack(w:Vec<usize>, v:Vec<i32>, cap:usize){ // choose item at most once
  n=w.len(); dp=vec![vec![0;cap+1];n+1];
  for i in 1..=n {for c in 0..=cap {
    dp[i][c]=dp[i-1][c];
    if w[i-1]<=c {
      dp[i][c]=dp[i][c].max(v[i-1]+dp[i-1][c-w[i-1]])
    }
  }}
  dp[n][cap]
}
// 1D 0/1: capacity descending. Unbounded: capacity ascending.
\end{lstlisting}
$\Theta(nW)$ pseudo-polynomial.

\subsectiontitle{Cut / Cycle Properties}
Cut $(S,V-S)$: edge crosses cut iff one endpoint in $S$, one outside. Light edge = minimum-weight crossing edge.

\textbf{Cut property:} if cut respects chosen edge set $A$, then a light crossing edge is safe.

\textbf{Cycle property:} for any cycle, a heaviest edge on that cycle is excluded from some MST. If weights distinct, the heaviest edge is in no MST.

fn max_flow(source:usize, sink:usize){

\sectiontitle{Randomized / Probabilistic Analysis}
Indicator: $X_A=\mathbf 1[A]$, $\mathbb E[X_A]=\Pr(A)$. Linearity: $E[\sum_i X_i]=\sum_i E[X_i]$.
No independence needed.

Hiring: $P(i\text{ new best})=1/i$, so $E[\#hires]=\sum_i1/i=H_n=\Theta(\log n)$. Hat-check: $P(i\text{ own hat})=1/n$, so $E[\#fixed]=1$.

\textbf{Birthday:} $\Pr[\text{all different among }q\text{ into }m]=\prod_{i=0}^{q-1}(1-i/m)\approx e^{-q(q-1)/(2m)}$; for 365 days, $q=23$ gives match prob $>1/2$.

\sectiontitle{Online Algorithms / Competitive Analysis / Experts}
Online: decide before future known.
Competitive ratio: $ALG(I)\le r\,OPT(I)+c$.
\textbf{Ski rental:} rent 1/day, buy B. Rent until paid B, then buy. If $T\le B$: ALG=T=OPT. If $T>B$: ALG<=B+B=2B=2OPT. Lower bound: adversary stops just before/after purchase; deterministic CR=2.
\subsectiontitle{Caching MC traps}
LRU evicts least-recently requested page. LFU evicts smallest request count; tie usually LRU. Neither has pointwise fewer misses than the other on every sequence. LRU is $k$-competitive for cache size $k$; no deterministic paging algorithm beats $k$ in worst-case competitive ratio.

\subsectiontitle{Experts / Weighted Majority}
$n$ experts, rounds $t=1..T$. Expert $i$ predicts; loss $\ell_{i,t}\in\{0,1\}$.
Weights $w_i=1$ initially. Let $W=\sum_iw_i$.

\begin{lstlisting}[aboveskip=0pt,belowskip=0pt]
for t in 1..=T {
  predict weighted majority of experts;
  observe true outcome;
  for expert i wrong this round {
    w[i] *= beta;        // 0 < beta < 1, often beta=1/2
  }
}
\end{lstlisting}

Weighted prediction: choose label with larger total weight.
Potential: $W_t=\sum_i w_i^{(t)}$.
If ALG makes mistake, wrong-weight $\ge W_t/2$, so
$W_{t+1}\le W_t/2\cdot\beta+W_t/2=W_t(1+\beta)/2$.
If best expert has $m^*$ mistakes, its weight is $\beta^{m^*}$, hence
$\beta^{m^*}\le W_T\le n\big((1+\beta)/2\big)^M$.
For $\beta=1/2$: $M\le 2.41(m^*+\log_2 n)$.
General randomized Hedge: choose expert with prob $w_i/W$; update $w_i\leftarrow w_i e^{-\eta \ell_{i,t}}$; regret $O(\sqrt{T\log n})$.
\subsectiontitle{Prophet Inequality}
Independent nonnegative $X_i$, known distributions. $M=\max_iX_i$, $z=E[M]$, threshold $\tau=z/2$. Accept first $X_i\ge\tau$. Guarantee $E[ALG]\ge z/2$. Proof stencil: $q=P[\forall i:X_i<\tau]$, $b_i=E[(X_i-\tau)^+]$. Value from threshold part $\ge\tau(1-q)$; overshoot part $\ge q\sum_i b_i$; combine to get $\ge z/2$.

Secretary: random order; observe first $n/e$ candidates, remember best rank seen, then hire first later candidate better than all observed. P(success) \approx 1/e

\workedq{P03}{Recursion tree: T(n)=8T(n/2)+Theta(n\^{}2), leaves dominate.}{
Level i: 8\^{} i nodes, size n/2\^{} i, work 8\^{} i*c*(n/2\^{} i)\^{}2 = c*n\^{}2*2\^{} i. Sum until lg n: c*n\^{}2*sum\_{i=0..lg n} 2\^{} i = Theta(n\^{}3). Leaves: 8\^{}(lg n)=n\^{}3. Hence Theta(n\^{}3).

\workedq{P04}{Recursive insertion sort: remove last element, sort prefix, insert last.}{

\workedq{P05}{Strengthened recurrence proof: split sizes 3n/5, 4n/5 plus linear work.}{

Check d[s]=0. For every edge (u,v) check triangle inequality d[v]<=d[u]+w(u,v); if violated reject. Build tight-edge graph E'=\{(u,v): d[v]=d[u]+w(u,v)\}. DFS/BFS from s in E'. Accept iff all vertices reached. Inequality gives d[v] <= every path length; tight paths give actual paths of length d[v], so equality.
}

\workedq{P34}{Bellman-Ford negative-cycle / altitude consistency with signed edge changes.}{
A bad closed tour is a directed cycle with nonzero total weight. Run Bellman-Ford on w and on -w from a super-source connected to all vertices by 0 edges. A negative cycle in w or in -w means some cycle has nonzero sum. If neither run detects, all cycles have sum 0. Runtime O(VE).
}

\workedq{P35}{Shortest-path weight transform MCQ: add constant vs multiply by positive constant.}{
Multiplying all edge weights by a>0 preserves every path ordering: w(P) becomes a*w(P). Adding a to every edge changes w(P) to w(P)+a*|P|, so paths with different numbers of edges can change order.
}

\workedq{P36}{Dijkstra choice proof: nonnegative weights only.}{
Maintain finalized set S. Pick outside vertex x with minimum tentative distance. Any s->x path first crosses from S to V-S at (u,v); length to v is at least d[u]+w(u,v)>=d[v]>=d[x] because w>=0. Thus d[x] is final. Negative edges break this.
}

\workedq{P37}{Bellman-Ford exact invariant and negative-cycle certificate.}{
After i full passes, d[v] is shortest path length among paths using at most i edges. Induct by last edge. Without negative cycles, shortest paths can be simple, so use at most V-1 edges. If a V-th pass still relaxes, there is a reachable negative cycle.
}

\workedq{P38}{MST over complete graph of points: connect all islands with n-1 routes.}{
Build complete undirected graph on the n points with edge weight squared Euclidean distance. Need connected graph with n-1 routes minimizing total cost = MST. Run Kruskal/Prim on O(n\^{}2) edges. Runtime O(n\^{}2 log n); output MST edges.
}

\workedq{P39}{MST cut/cycle proof: min crossing edge safe, max cycle edge unsafe.}{
Cut: add lightest crossing edge e to an MST not using it; cycle contains another crossing edge f with w(f)>=w(e); swap f->e. Cycle: if unique heaviest edge e of a cycle is in MST, remove it; another cycle edge reconnects the cut cheaper, contradiction.
}

\workedq{P40}{Minimum-weight feedback edge set in undirected weighted graph.}{
Need remove edges so remaining graph is acyclic. Equivalent: keep an acyclic subset of maximum total weight. In connected graph, max kept acyclic set is a maximum spanning tree. Run Kruskal in descending weights, keep tree edges, remove all others. Correct because any acyclic connected kept set has at most V-1 edges and maximum spanning tree maximizes kept weight.
}

\workedq{P41}{Airport + roads connectivity: cities can connect by roads or both having airports.}{
Add super-node r. For every city v, add edge (r,v) of cost airport cost c[v]. Keep original road edges with costs w[e]. Run MST on augmented graph. Chosen original edges = roads; chosen r-edges = airports. Super-node models free travel between airports. Runtime O((E+V)log(E+V)).
}

\workedq{P42}{Edge belongs to at least one MST: linear-time query for edge e=(a,b).}{
Let W=w(e). Run DFS/BFS from a using only edges with weight <W. If b reached, then e is heaviest on a strictly lighter cycle, so no MST contains it. If not reached, take component S of a in subgraph w<W; every edge crossing (S,V-S) has weight >=W, and e crosses, so e is lightest for this cut. Thus some MST contains e. Runtime O(V+E).
}

\workedq{P43}{Unique MST vs many MSTs: detect swappable equal-weight non-tree edge.}{
Compute one MST T. For every non-tree edge e=(u,v), find maximum edge weight M on the unique path u->v in T. If w(e)=M, swap e with a max edge on that path to get another MST: MANY. If all w(e)>M, no equal swap exists: ONE. Naive runtime O(EV); faster with LCA if needed.
}

\workedq{P44}{Connected graph with n vertices and n edges: count spanning trees in O(n).}{
Connected with |E|=|V| has exactly one cycle. Find cycle length c by DFS back edge or repeatedly delete degree-1 leaves until the cycle remains. Every spanning tree is obtained by deleting exactly one cycle edge; deleting non-cycle edge disconnects. \ans number of spanning trees = c, hence between 3 and n for simple graph.
}

\workedq{P45}{Tree diameter: output farthest pair of nodes in binary tree.}{
Postorder returns (height, deepest, bestLen, bestPair). At node u, candidate crossing pair is deepest node in left subtree to deepest node in right subtree with length hL+hR+2 (handle empty child). Best is max of left best, right best, crossing. Runtime O(n).
}

\workedq{P46}{Longest simple path constrained to pass through the root.}{
Compute heights of left and right subtrees. If height counts edges and empty height is -1, answer is h(root.left)+h(root.right)+2. The path through root is maximized by independently choosing deepest descendant on each side. Runtime O(n).
}

\workedq{P47}{Reconstruct binary tree from inorder and postorder traversals.}{
Root is last element of postorder. Find root in inorder; left/right inorder blocks determine subtree sizes. Split postorder into corresponding left/right blocks, recurse. Base empty list. Runtime O(n) with value-to-inorder-index map. Preorder+inorder similar; preorder+postorder alone is not unique.
}

\workedq{P48}{Travel passes DP: cover sorted travel days with 1/7/30-day passes.}{
Days d[1]<...<d[n], lengths L, costs c[L]. dp[i] = min cost to cover days i..n. next(i,L)=first j with d[j]>=d[i]+L. dp[n+1]=0, dp[i]=min over L of c[L]+dp[next(i,L)]. Fill decreasing i. \ans dp[1]. Trap: pass bought on day d[i] covers days <d[i]+L.
}

\workedq{P49}{Weighted interval scheduling: choose non-overlapping intervals of max weight.}{
Sort by finish. p(i)=last interval ending <=start[i]. dp[i]=best using first i intervals. dp[0]=0, dp[i]=max(dp[i-1], w[i]+dp[p(i)]). \ans dp[n]. Runtime O(n log n) with binary search for p.
}

\workedq{P50}{Cardinality-constrained weighted intervals: choose at most k disjoint intervals.}{
Sort by finish and compute p(i). dp[i][j]=best value using first i intervals and at most j intervals. Base dp[0][j]=dp[i][0]=0. Recurrence dp[i][j]=max(dp[i-1][j], w[i]+dp[p(i)][j-1]). \ans dp[n][k]. Runtime O(nk+n log n).

\workedq{P51}{Interval scheduling greedy: max number of mutually disjoint intervals.}{
Sort by increasing finish time. Repeatedly take first interval compatible with current last finish. Exchange proof: earliest-finishing interval e can replace first interval of any optimum without reducing feasibility. Then recurse on intervals starting after e.
}

\workedq{P52}{Divide farm into exactly 3 contiguous nonempty parts minimizing largest sum.}{
Input A[1..n], A[i] positive and A[i]<=1000. Need cuts i<j minimizing max(sum 1..i, sum i+1..j, sum j+1..n). Binary-search answer M = allowed max child area. Bounds: L=max A[i] (cannot put A[i] anywhere if M<A[i]), R=sum A[i] (one part contains all). Feasible(M): scan left->right with cur=0, parts=1; if cur+x<=M then cur+=x, else parts++, cur=x. Reject if parts>3. Why greedy works: for fixed M, greedy makes each part as long as possible, so after t parts it has covered at least as much prefix as any valid t-part partition; hence greedy uses the minimum possible \#parts. Therefore parts>3 => impossible. If parts<=3 => possible for exactly 3: n>=3 and all A[i]>0, so split existing nonempty parts until there are 3; splitting only decreases part sums. Predicate monotone: bigger M cannot need more parts. Binary search smallest feasible M. Runtime Feasible=O(n), R<=1000n => O(n log(1000n))=O(n log n). No direct sum(a,b) needed; cur is computed during scan. Recover cuts: rerun greedy at M*, then split if fewer than 3 parts, output the two cut indices.
}

\workedq{P53}{Split array into k contiguous parts minimizing maximum load.}{
Same as P52 with k instead of 3. L=max a[i], R=sum a[i]. Feasible(M) greedily computes the minimum number q of parts needed with each part sum<=M. Feasible iff q<=k; if exact k required and n>=k, split parts until k. Binary search smallest feasible M. Runtime O(n log R); if a[i] polynomially bounded, O(n log n).
}

\workedq{P54}{Short grid walk with value and move limit: moves right/up/diag, at most 3n/2 moves.}{
dp[i][j][ell] = max value reaching cell (i,j) in exactly ell moves. Base dp[0][0][0]=v(0,0). Transition from predecessors (i-1,j), (i,j-1), (i-1,j-1) with ell-1, add v(i,j). \ans max over ell<=floor(3n/2) of dp[n][n][ell]. Runtime O(n\^{}3).

\workedq{P55}{Rod cutting / unbounded segmentation: maximize revenue over cuts summing to n.}{
r[0]=0. For length j=1..n: r[j]=max over 1<=i<=j of p[i]+r[j-i]. This chooses leftmost piece length i and solves remainder optimally. Runtime O(n\^{}2), space O(n). Store argmax to reconstruct cuts.

\workedq{P56}{Matrix-chain multiplication: choose parenthesization minimizing scalar multiplications.}{
Matrices A[i] have dimensions p[i-1] x p[i]. m[i,i]=0. For length ell=2..n: m[i,j]=min over i<=k<j of m[i,k]+m[k+1,j]+p[i-1]*p[k]*p[j]. \ans m[1,n]. Runtime O(n\^{}3), space O(n\^{}2).

\workedq{P57}{LCS of two strings: subsequence, not substring.}{

\workedq{P58}{Longest increasing subsequence: standard DP and faster tails version.}{
DP: L[j]=1+max\{L[i]: i<j and A[i]<A[j]\}, empty max 0; answer max\_j L[j], O(n\^{}2). Tails: maintain smallest possible tail for each length; for x, lower\_bound first tail >= x and replace. Length of tails = LIS length, O(n log n).

\workedq{P59}{Longest modular increasing subsequence: increasing values and sum divisible by 3.}{
r[j,b] = best length of increasing subsequence ending at j with sum mod 3 equal b. Initialize r[j,A[j] mod 3]=1. For all i<j with A[i]<A[j] and all b: r[j,(b+A[j]) mod 3]=max(r[j,(b+A[j]) mod 3], r[i,b]+1). \ans max\_j r[j,0]. Runtime O(n\^{}2).

\workedq{P60}{0/1 knapsack: each item used at most once, capacity W.}{
dp[i][w] = best value using first i items with capacity w. dp[0][w]=0. If weight[i]>w, skip; else dp[i][w]=max(dp[i-1][w], value[i]+dp[i-1][w-weight[i]]). Runtime O(nW) pseudo-polynomial. 1D update w decreasing.

\workedq{P61}{3-way equal partition: split positive integers into 3 subsets each sum W/3.}{
If W mod 3 != 0, NO. Let target S=W/3. dp[i][x][y] = possible after first i items with subset1 sum x, subset2 sum y; subset3 implicit. Transition: put item i in subset1 if x>=w[i], subset2 if y>=w[i], or subset3. \ans dp[n][S][S]. Runtime O(nS\^{}2).

\workedq{P63}{Coin change / minimum number of coins, unlimited copies.}{
dp[0]=0, dp[x]=INF initially. For amount x=1..W: dp[x]=1+min over c<=x of dp[x-c]. \ans dp[W] or impossible if INF. Runtime O(W*\#coins). For number of ways, loop coins outside and amounts increasing.

\workedq{P64}{Optimal BST DP with key probabilities / search frequencies.}{
e[i,j] = min expected cost for keys i..j, w[i,j] = total probability/frequency in interval. Choose root r: e[i,j]=min over r=i..j of e[i,r-1]+e[r+1,j]+w[i,j]. Empty interval cost 0. Fill by increasing length. Runtime O(n\^{}3).

\workedq{P66}{Ski rental competitive analysis: rent price 1, buy price B.}{
Algorithm: rent until total rent reaches B, then buy. If T<=B, ALG=T=OPT. If T>B, ALG<=B+B=2B=2OPT. Lower bound for deterministic algorithms: adversary stops just before algorithm buys or continues just after; no deterministic ratio <2.

\workedq{P67}{Paging/caching LRU: cache size k, prove k-competitive.}{
Partition requests into phases with at most k distinct pages; new phase starts when (k+1)-st distinct page appears. LRU has at most k misses per phase. OPT has at least 1 miss per completed phase because phase plus next first request contains k+1 distinct pages. Hence LRU <= k*OPT + O(k).
}

\workedq{P68}{Weighted majority / experts: bound mistakes against best expert.}{
Weights start 1; after wrong prediction multiply expert weight by beta. If algorithm makes a mistake, wrong side has weight >=W/2, so W'<=W(1+beta)/2. Best expert with m* mistakes has weight beta\^{}(m*) <= W[T] <= n((1+beta)/2)\^{} M. Solve for M.

\workedq{P69}{Secretary problem: random order, choose maximum candidate online.}{
Reject first r=floor(n/e), remember best sample. Then accept first candidate better than all seen. Success if global best appears at position i>r and best among first i-1 was in sample. Probability sum\_{i=r+1..n} (1/n)*(r/(i-1)) approx (r/n)ln(n/r) approx 1/e.

\workedq{P70}{Online bipartite matching: greedy and ranking guarantees.}{
Greedy matches each arriving right vertex to any free neighbor. Each ALG edge can be charged by at most two OPT edges, so ALG >= OPT/2. Ranking: pick random priority order on offline left vertices; each arrival chooses highest-priority free neighbor; expected value (1-1/e)OPT.

\workedq{P71}{Prophet inequality: threshold at half expected maximum.}{
Independent nonnegative X[i], known distributions. Let M=max\_i X[i], z=E[M], threshold tau=z/2. Accept first X[i]>=tau. Guarantee E[ALG]>=z/2. Proof uses q=P[all i: X[i]<tau] and overshoots (X[i]-tau)\^{}+; threshold contribution tau(1-q) plus overshoot contribution q*sum\_i E[(X[i]-tau)\^{}+].
}

\microq{Growth order}{log n < log(n!)=Theta(n log n) < n log n < n2 < n3 < 2n < 3n < n! < nn. Also log\^{}k n=o(n\^{}eps), n\^{}k=o(c\^{}n).}
\microq{Weird growth}{(log n)\^{}(log n)=n\^{}(log log n): super-polynomial, sub-exponential. n\^{}(sqrt n)=e\^{}(sqrt n log n) < c\^{}n but > n\^{}k.}
\microq{Factorial estimate}{Stirling: n!=(Theta(sqrt n))(n/e)\^{}n. For MCQ usually enough: log(n!)=sum log i=Theta(n log n).}
\microq{Recurrence T(n-1)}{T(n)=T(n-1)+f(n) => sum f. T(n)=aT(n-1)+1 with a>1 => Theta(a\^{}n).}
\microq{Recurrence sqrt}{T(n)=aT(sqrt n)+f(n). Put m=log n, S(m)=aS(m/2)+f(2\^{}m). Example T(n)=4T(n\^{}(1/4))+log n => S(m)=4S(m/4)+m=Theta(m log m).}
\microq{Unequal split AB}{T(n)=sum a\_i T(b\_i n)+Theta(n\^{}d). Find p with sum a\_i b\_i\^{}p=1. p>d => Theta(n\^{}p); p=d => Theta(n\^{}p log n); p<d => Theta(n\^{}d).}
\microq{AB examples}{T(n)=T(n/3)+T(n/2)+n: p<1, answer Theta(n). If alpha\^{}d+beta\^{}d=1 and work n\^{}d, answer Theta(n\^{}d log n).}
\microq{Strengthening recipe}{If substitution gives target + bad lower term, prove stronger upper T(n)<=main-lowerTerm. LowerTerm must create slack under recursive shrink.}
\microq{Master trap}{Master only for aT(n/b)+f(n), fixed a,b. Not for T(n-1), T(sqrt n), unequal splits unless transformed/AB.}
\microq{Asymptotic proof}{To show f=O(g): find c,n0. To show Omega: find c,n0 lower. Theta = both. For sums, dominant term wins if positive.}
\microq{Stack trace}{Push in array order then pop gives reversed order. Queue enqueue then dequeue gives same order. Underflow only if pop/dequeue empty.}
\microq{Heap indices}{1-index: parent=floor(i/2), left=2i, right=2i+1. 0-index: parent=(i-1)/2, left=2i+1, right=2i+2.}
\microq{Heap trace}{BuildHeap calls Heapify from floor(n/2) downto 1. In max-heap, root is max; heapsort swaps root with end, shrinks heap, so final array ascending.}
\microq{Stable/in-place MCQ}{Insertion stable/in-place. Merge stable/not in-place. Heap in-place/not stable. Random quicksort in-place-ish/not stable, expected n log n, worst n2.}
\microq{Comparison sort lower}{Only comparison sorts have Omega(n log n) lower bound. Counting/radix can beat it using key structure.}
\microq{Binary lower bound}{lower-bound returns first i with A[i]>=x. Inv: answer in [l,r]. If A[mid]>=x then r=mid else l=mid+1.}
\microq{Binary upper bound}{upper-bound returns first i with A[i]>x. Count <=x is upper\_bound(x); count <x is lower\_bound(x).}
\microq{Floor/ceil search}{floor x = last <=x = upper\_bound(x)-1. ceil x = first >=x = lower\_bound(x). Check sentinel when index outside array.}
\microq{BST height MCQ}{Any binary tree with n nodes has height at least floor(log2 n). A BST can be height n-1. Search worst O(h).}
\microq{BST duplicates}{State convention. If left < root and right >= root, validation interval should be (lo,key) on left and [key,hi) or adapted on right.}
\microq{Adjacency matrix/list}{List space Theta(V+E), scan neighbors O(deg). Matrix space Theta(V2), edge lookup O(1), scanning neighbors O(V).}
\microq{Transpose graph}{For directed G, build G\^{}T by reversing every edge in adjacency lists. Runtime Theta(V+E). Used in Kosaraju.}
\microq{Connected components}{Undirected: run BFS/DFS from every unvisited vertex; each search outputs one component. Runtime Theta(V+E).}
\microq{Tree tests}{Undirected graph is tree iff connected and E=V-1 iff acyclic and E=V-1 iff connected and acyclic. Forest with c comps has E=V-c.}
\microq{Bipartite test}{BFS color by parity distance. If edge joins same color => not bipartite; else partition is bipartition. Runtime Theta(V+E).}
\microq{DAG facts}{Every nonempty DAG has a source and sink. Repeatedly removing sources gives topo order. Directed cycle means no topo order.}
\microq{Topo uniqueness}{Topo order is unique iff every consecutive pair in the produced order has an edge between them in that direction. Otherwise swap adjacent incomparable vertices.}
\microq{DFS edge classes}{Directed DFS: tree, back(to gray ancestor), forward(to descendant), cross(other). Undirected DFS has only tree/back edges.}
\microq{SCC condensation DP}{Condense SCCs to DAG. To solve reachability/order DP between components, solve on condensation, not original cycles.}
\microq{MST distinct weights}{If all edge weights are distinct, MST is unique. Proof: if two MSTs differ, take lightest edge in symmetric difference and use cut/cycle exchange.}
\microq{MST equal weights}{Equal weights may give multiple MSTs, but Kruskal/Prim still return an MST. Tie-breaking changes tree, not optimum weight.}
\microq{MST negative weights}{Negative edges are fine; cut/cycle properties do not require nonnegative weights. Kruskal still works.}
\microq{MST vs SPT}{MST minimizes total tree weight; shortest-path tree preserves distances from source. Neither implies the other.}
\microq{MST transform}{Add constant c to every edge: MST unchanged because every spanning tree has V-1 edges. Multiply by positive c: unchanged. Negative multiply turns min into max.}
\microq{Bottleneck tree}{Every MST is a minimum-bottleneck spanning tree: its largest edge weight is as small as possible. Converse need not hold.}
\microq{Bridge/cycle MST}{Every bridge is in every spanning tree/MST. A heaviest edge can still be in MST if it is a bridge; cycle property only applies to cycles.}
\microq{Dijkstra zeros}{Zero-weight edges OK. Negative edges not OK because finalized vertex might later improve.}
\microq{Unreachable vertices}{BFS/Dijkstra/BF leave d=INF and parent=NIL. Do not claim negative cycle affects answer unless reachable from source.}
\microq{BF any negative cycle}{To detect any negative cycle anywhere, add super-source s to every vertex with weight 0, then run Bellman-Ford.}
\microq{Difference constraints}{Constraint x\_v <= x\_u + w(u,v) becomes edge u->v weight w. Feasible iff no negative cycle. Distances from super-source give solution.}
\microq{Arbitrage}{Exchange rates r(u,v). Product cycle >1 iff sum of weights -log r(u,v) on cycle <0. Detect with Bellman-Ford.}
\microq{Shortest cycle through edge}{For directed edge e=(u,v), shortest cycle using e = w(e)+shortest path v->u after removing/allowing e carefully.}
\microq{Number of shortest paths}{During BFS/Dijkstra: if better dist, set count=count[u]; if equal dist, add count[u]. Need process in nondecreasing dist.}
\microq{Even/odd path}{Duplicate states (v,parity). Edge toggles parity. BFS/Dijkstra on product graph; answer dist[(t,desiredParity)].}
\microq{Mod-state shortest path}{For path length/cost mod k, use states (v,r). Edge updates r. Then run BFS/Dijkstra depending on weights.}
\microq{One special edge/state}{If one coupon/free/expensive action allowed, states (v,used). Normal transitions keep used; special transition flips 0->1.}
\microq{Path reconstruction}{Store parent when relaxing/discovering. Reconstruct by following parent from target to source and reversing. For DP store argmin choice.}
\microq{Residual edge meaning}{Forward residual cap = unused capacity. Backward residual cap = current flow that can be canceled. Augmenting may undo earlier choices.}
\microq{Augment bottleneck}{Path push amount = min residual capacity on path. At least one edge becomes saturated, but later back edges can unsaturate it.}
\microq{Undirected capacity}{For undirected edge capacity c, common reduction uses two opposite arcs cap c; opposite flows can be canceled to respect net capacity. For edge-disjoint, use a single cap-1 gadget.}
\microq{Min edge cut}{Set every edge capacity 1. Min s-t cut value = minimum number of edges whose removal separates s,t.}
\microq{Min vertex cut}{Split each nonterminal v into vin->vout cap 1, original edges go out->in with INF. Min cut value = min internal vertices to remove.}
\microq{Interval DP shape}{If answer for interval [i,j] depends on choosing split/root k inside, states are intervals, fill by increasing length. Runtime often O(n3).}
\microq{Grid DP shape}{If moves only right/down/up-diag in acyclic direction, dp[cell]=best over predecessor cells. Fill in topological coordinate order.}
\microq{Exact-k DP}{Add dimension k. dp[i][k]=best using prefix/items with exactly k choices. For at most k, answer max over <=k or carry skip transitions.}
\microq{Feasibility DP}{Boolean dp means possible, transition OR. Optimization dp means best value, transition min/max. Do not mix proof language.}
\microq{Subset sum}{dp[x]=possible sum x. For each item w, update x decreasing: dp[x]|=dp[x-w]. Decreasing prevents using item twice.}
\microq{Unbounded loop order}{Unbounded coin/rod: update amount increasing so same item can be reused. 0/1: capacity decreasing.}
\microq{Substring vs subsequence}{Substring must be contiguous: if chars equal dp[i][j]=dp[i-1][j-1]+1 else 0. Subsequence uses max(skip left, skip right).}
\microq{DP with sorted choices}{Sort by natural endpoint/time/value first. Then predecessor p(i) via binary search or two pointers; recurrence usually take/skip.}
\microq{Solution reconstruction DP}{When taking max/min, store arg choice. After table filled, walk choices backward/recursively to output selected objects.}
\microq{Pseudo-polynomial}{O(nW) is polynomial in numeric value W, not in input bitlength log W. Say pseudo-polynomial if weights are input numbers.}
\microq{Greedy fail signal}{If choices have weights/profits interacting with future capacity, suspect DP not greedy. Example: 0/1 knapsack not ratio-greedy.}
\microq{Interval scheduling proof}{Earliest finishing interval is safe: replace first interval of any optimal solution by it; leaves at least as much room for the rest.}
\microq{Sweep tie rules}{For half-open [s,e), process end before start at same coordinate. For closed intervals, process start before end if touching counts as overlap.}
\microq{Binary answer bounds}{Pick L definitely infeasible or minimal possible, R definitely feasible or maximal possible. State why bounds contain optimum.}
\microq{Monotone predicate}{Minimize maximum: if M feasible, larger M feasible. Maximize minimum: if M feasible, smaller M feasible. Adjust binary search direction.}
\microq{Random indicators}{Linearity of expectation does not require independence. Define indicator for each local event, compute its probability, sum.}
\microq{Union bound}{P(any bad event) <= sum P(bad\_i). Combine with expectation: P[X>=1] <= E[X].}
\microq{Markov}{For nonnegative X: P[X>=a] <= E[X]/a. Useful to turn expected bad count into existence/high-prob bound.}
\microq{Pairwise independence}{For expected collision count, pairwise independence/universal hashing is enough; no need full independence over all keys.}
\microq{Randomized worst case}{Randomized quicksort has expected O(n log n), but a particular unlucky pivot sequence still gives O(n2). Expected != worst-case.}
\microq{Online ratio form}{For minimization, ALG(I) <= r OPT(I)+c where c independent of input length. For maximization, usually ALG >= OPT/r - c.}
\microq{Adversary lower bound}{To prove no online alg beats r, adaptively choose continuation after seeing ALG action; compare ALG cost/value with offline OPT on same sequence.}
\microq{Index convention}{Before writing recurrence, define whether intervals are [i,j], [i,j), or prefix first i items. Most off-by-one errors come from this.}
\microq{Travel pass endpoint}{Pass of length L bought on day d covers integer days d..d+L-1. Previous uncovered day must be < d, or next index has day > d+L-1.}
\microq{INF choice}{For reductions, INF can be any number larger than max possible finite objective/cut, e.g. sum of all finite capacities+1.}
\microq{Certificate habit}{For optimization, mention certificate: MST edges total weight, shortest-path distances+parents, maxflow+mincut, DP table+choices.}
\microq{Runtime habit}{State graph size after reduction. If time-expanded/layered graph has T layers, multiply V,E by T before applying BFS/Dijkstra/flow.}
\microq{Proof ending}{Always finish: algorithm returns feasible solution; by lemma/invariant no better solution exists; hence optimal. Then runtime.}
